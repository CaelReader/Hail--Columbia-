#USA: philippines subjugated
usfp_philippines.1 = {
    type = country_event
    placement = scope:philippines_scope.capital
    event_image = {
        video = "asia_westerners_arriving"
    }

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

    icon = "gfx/interface/icons/event_icons/waving_flags_usa_philippines.dds"

    title = usfp_philippines.1.t
    desc = usfp_philippines.1.d
    flavor = usfp_philippines.1.f
    dlc = dlc_usfp
    duration = 3
    cooldown = { days = never_fire_again_modifier_time } #Once

    immediate = {

    }

    #begin tutelage
    option = {
        name = usfp_philippines.1.a
        default_option = yes
        scope:philippines_scope = {
            add_modifier = usfp_philippines_military_government_modifier
            activate_law = law_type:law_colonial_administration
            activate_law = law_type:law_autocracy
            set_ruling_interest_groups = { ig_armed_forces }
            change_subject_type = subject_type_colony
            trigger_event = {
                id = usfp_philippines.2
                days = 5
            }
            create_character = {
                is_general = yes
                interest_group = ig:ig_armed_forces
                culture = cu:yankee
                religion = protestant
                ideology = ideology:ideology_modernizer_leader
                ruler = yes
                ig_leader = yes
                on_created = {
                    add_trait = special_character_colonial_military
                }
            }
        }
        create_unidirectional_truce = {
            country = scope:philippines_scope
            months = 60
        }
        ai_chance = {
            base = 100
        }
    }

    #release as independent
    option = {
        name = usfp_philippines.1.b
        scope:philippines_scope = {
            make_independent = yes
            activate_law = root.active_law:lawgroup_distribution_of_power.type
            activate_law = root.active_law:lawgroup_governance_principles.type
            create_political_lobby = {
                type = lobby_pro_country
                target = root
                add_interest_group = ig:ig_intelligentsia
            }
        }
        if = {
            limit = {
                is_power_bloc_leader = yes
            }
            power_bloc = {
                add_leverage = {
                    target = c:PHI
                    value = 300
                }
            }
        }
        change_relations = {
            country = scope:philippines_scope
            value = 50
        }
        create_bidirectional_truce = {
            country = scope:philippines_scope
            months = 60
        }
        custom_tooltip = {
            text = close_philippines_je
            set_variable = rejected_philippines_tutelage
        }
        ai_chance = {
            base = 0
        }
    }

    #reject JE
    option = {
        name = usfp_philippines.1.c
        ig:ig_intelligentsia = {
            add_modifier = {
                name = usfp_rejected_philippine_tutelage_ig_modifier
                years = 4
            }
        }
        change_relations = {
            country = c:PHI
            value = -25
        }
        custom_tooltip = {
            text = close_philippines_je
            set_variable = rejected_philippines_tutelage
        }
        ai_chance = {
            base = 0
        }
    }
}

    desc = usfp_philippines.2.d
    flavor = usfp_philippines.2.f
    dlc = dlc_usfp
    duration = 3
    cooldown = { days = never_fire_again_modifier_time } #Once

    immediate = {
        overlord = {
            save_scope_as = usa_scope
        }
    }

    #assert independence
    option = {
        name = usfp_philippines.2.a
        default_option = yes
        highlighted_option = yes
        trigger = {
            any_political_movement = {
                OR = {
                    is_political_movement_type = movement_radical
                    is_political_movement_type = movement_liberal
                }
            }
        }
        show_as_unavailable = {
            always = yes
        }
        ordered_political_movement = {
            order_by = political_movement_support
            position = 0
            limit = {
                OR = {
                    is_political_movement_type = movement_radical
                    is_political_movement_type = movement_liberal
                }
            }
            add_modifier = {
                name = usfp_philippine_revolt_modifier
                years = 5
            }
        }
        add_modifier = {
            years = 1
            name = usfp_imminent_revolution_modifier
        }
        add_liberty_desire = 25
        c:USA = {
            post_notification = usfp_philippines_nationalist_revolt
        }
        ai_chance = {
            base = 100
        }
    }

    #accept tutelage
    option = {
        name = usfp_philippines.2.b
        add_liberty_desire = -25
        add_loyalists = {
            value = medium_radicals
            interest_group = ig:ig_intelligentsia
        }
        c:USA = {
            post_notification = usfp_philippines_accept_tutelage
        }
        ai_chance = {
            base = 1
        }
    }

    after = {
        show_as_tooltip = {
            add_modifier = usfp_philippines_military_government_modifier
            activate_law = law_type:law_colonial_administration
            activate_law = law_type:law_autocracy
            set_ruling_interest_groups = { ig_armed_forces }
            change_subject_type = subject_type_colony
        }
    }

    cancellation_trigger = {
        NOT = {
            is_subject_of = c:USA
        }
    }
}

usfp_philippines.3 = {
    type = country_event
    placement = scope:moro_state
    event_image = {
        video = "ip3_angry_mob_raiding_minorities"
    }

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

    icon = "gfx/interface/icons/event_icons/event_fire.dds"

    title = usfp_philippines.3.t
    desc = usfp_philippines.3.d
    flavor = usfp_philippines.3.f
    dlc = dlc_usfp
    duration = 3
    cooldown = { days = never_fire_again_modifier_time } #Once

    trigger = {
        c:USA ?= root
        c:PHI ?= {
            is_subject_of = root
            is_diplomatic_play_committed_participant = no
            has_revolution = no
            any_scope_state = {
                state_region = {
                    is_homeland = cu:moro
                }
            }
        }
        is_involved_in_journal_entry = je_usfp_philippine_tutelage
    }

    immediate = {
        c:PHI ?= {
            random_scope_state = {
                limit = {
                    state_region = {
                        is_homeland = cu:moro
                    }
                }
                save_scope_as = moro_state
            }
        }
    }

    #crack down
    option = {
        name = usfp_philippines.3.a
        default_option = yes
        c:PHI ?= {
            every_scope_state = {
                limit = {
                    state_region = {
                        is_homeland = cu:moro
                    }
                }
                add_devastation = 20
            }
            add_radicals = {
                culture = cu:moro
                value = very_large_radicals
            }
            post_notification = moro_rebellion_crackdown
        }
        ai_chance = {
            base = 100
        }
    }

    #abandon those areas
    option = {
        name = usfp_philippines.3.b
        c:PHI ?= {
            every_scope_state = {
                limit = {
                    state_region = {
                        is_homeland = cu:moro
                    }
                }
                set_state_type = unincorporated
                add_modifier = {
                    name =  usfp_tribal_state_rule_modifier
                    years = 10
                }
            }
            post_notification = moro_rebellion_withdraw
        }
        ai_chance = {
            base = 0
        }
    }

}
