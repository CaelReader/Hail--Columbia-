#New England Emigrant Aid Society
usfp_bleeding_kansas.1 = {
	type = country_event

	icon = "gfx/interface/icons/event_icons/event_protest.dds"
	event_image = {
		video = "europenorthamerica_rich_and_poor"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	title = usfp_bleeding_kansas.1.t
	desc = usfp_bleeding_kansas.1.d
	flavor = usfp_bleeding_kansas.1.f
	dlc = dlc_usfp
	placement = scope:kansas_scope

	duration = 3

	trigger = {
		NOT = {
			has_variable = new_england_emigrant_aid_society
		}
	}

	immediate = {
		s:STATE_KANSAS.region_state:USA = {
			save_scope_as = kansas_scope
		}
		set_variable = new_england_emigrant_aid_society
	}

	#good
	option = {
		name = usfp_bleeding_kansas.1.a
		scope:kansas_scope = {
			custom_tooltip = {
				text = usfp_bleeding_kansas.1.a.tt
				create_pop = {
					culture = yankee
					size = 4000
				}
			}
			add_radicals_in_state = {
				culture = cu:yankee
				value = 0.02
			}
		}
		custom_tooltip = {
			text = usfp_bleeding_kansas_swing_free
			change_variable = {
				name = bleeding_kansas_var
				add = 5
			}
		}
	}

	#bad
	option = {
		name = usfp_bleeding_kansas.1.b
		scope:kansas_scope = {
			custom_tooltip = {
				text = usfp_bleeding_kansas.1.a.tt
				create_pop = {
					culture = yankee
					size = 2000
				}
			}
			add_radicals_in_state = {
				culture = cu:dixie
				value = 0.02
			}
		}
		ig:ig_intelligentsia = {
			add_modifier = {
				name = ig_abolitionists_mad
				months = short_modifier_time
			}
		}
	}
}
