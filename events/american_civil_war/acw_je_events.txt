namespace = acw_je_events

# ACW startup
acw_je_events.1 = {
	type = country_event
	event_image = {
		video = "europenorthamerica_american_civil_war"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = {
		first_valid = {
			triggered_desc = {
				trigger = {
					has_variable = csa_will_spawn_var
				}
				desc = acw_je_events.1.t1
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = acw_je_events.1.t2
			}
		}
	}
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					has_variable = csa_will_spawn_var
				}
				desc = acw_je_events.1.d1
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = acw_je_events.1.d2
			}
		}
	}
	flavor = {
		first_valid = {
			triggered_desc = {
				trigger = {
					has_variable = csa_will_spawn_var
				}
				desc = acw_je_events.1.f1
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = acw_je_events.1.f2
			}
		}
	}

	duration = 3

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		# triggered by je_acw_countdown completion
	}

	immediate = {
		if = {
			limit = {
				has_variable = csa_will_spawn_var
			}
			c:CSA ?= {
				save_scope_as = csa_scope	
				if = {
					limit = {
						NOT = { has_law_or_variant = law_type:law_legacy_slavery }
					}
					activate_law = law_type:law_legacy_slavery
				}	

				# Re-enslave pops if the ACW happens after slavery has been abolished
				if = {
					limit = {
						c:USA = {
							has_variable = slavery_recently_abolished
						}						
					}
					
					every_scope_state = {
						enslave_discriminated_farm_workers = yes
					}					
				}
				every_scope_state = {
					state_region = {
						add_claim = root
					}
				}
				if = {
					limit = { 
						NOT = {
							has_law_or_variant = law_type:law_presidential_republic
						}
					}
					activate_law = law_type:law_presidential_republic
				}
				add_declared_interest = region_england
				add_declared_interest = region_france
				activate_law = law_type:law_agrarianism
				activate_law = law_type:law_free_trade
				activate_law = law_type:law_national_supremacy
				activate_law = law_type:law_tenant_farmers
				activate_law = law_type:law_migration_controls
				set_ruling_interest_groups = {
					ig_landowners
				}
				add_modifier = {
					name = usfp_slave_power_modifier
					years = 20
				}
			}
		}
		else = {
			c:FSA ?= {
				save_scope_as = fsa_scope

				every_scope_state = {
					state_region = {
						add_claim = root
					}
				}				
				
				if = {
					limit = { 
						NOT = {
							has_law_or_variant = law_type:law_presidential_republic
						}
					}
					activate_law = law_type:law_presidential_republic
				}
				if = {
					limit = {
						NOT = { has_law_or_variant = law_type:law_slavery_banned }
					}
					activate_law = law_type:law_slavery_banned
				}

				trigger_event = {
					id = usfp_acw.8
					days = 10
				}
			}
		}
		set_global_variable = {
			name = american_civil_war
			value = yes
		}
		# random_diplomatic_play = {
		# 	limit = {
		# 		OR = {
		# 			AND = {
		# 				exists = scope:fsa_scope
		# 				any_scope_play_involved = {
		# 					this = scope:fsa_scope
		# 				}
		# 			}
		# 			AND = {
		# 				exists = scope:csa_scope
		# 				any_scope_play_involved = {
		# 					this = scope:csa_scope
		# 				}
		# 			}
		# 		}
		# 		# any_scope_play_involved = {
		# 		# 	OR = {
		# 		# 		this = scope:csa_scope
		# 		# 		this = scope:fsa_scope
		# 		# 	}
		# 		# }
		# 	}
		# 	remove_war_goal = {
		# 		who = initiator
		# 		war_goal = annex_country
		# 	}
		# 	add_war_goal = {
		# 		holder = initiator
		# 		type = revoke_all_claims
		# 		target_country = target				
		# 		primary_demand = yes
		# 	}			
		# }
		
		if = {
			limit = { exists = c:CSA }
			add_journal_entry = {
				type = je_acw_war
				target = c:CSA
			}
			c:CSA = {
				set_variable = acw_slave_nation_var
			}

			# historical Confederates move to CSA
			every_scope_character = {
				limit = {
					has_variable = historical_csa
					NOR = {
						is_ruler = yes
						is_heir = yes
					}
				}
				transfer_character = c:CSA
			}
		}
		else_if = {
			limit = { exists = c:FSA }
			add_journal_entry = {
				type = je_acw_war
				target = c:FSA
			}
			set_variable = acw_slave_nation_var
		}
		# set_variable = union_side_acw
	}

	option = {
		name = acw_je_events.1.a
		default_option = yes
		# if = {
		# 	limit = {
		# 		exists = c:CSA
		# 	 }
		# 	 custom_tooltip = acw_war_support_tooltip
		# }
		custom_tooltip = {
			text = je_usfp_underground_railroad_enhanced_effect
		}
	}
	option = {
		name = acw_je_events.1.b
		trigger = { is_player = yes }
		if = {
			limit = { exists = c:CSA }
			play_as = c:CSA
			custom_tooltip = {
				text = je_usfp_underground_railroad_enhanced_effect
			}
		}
		else_if = {
			limit = { exists = c:FSA }
			play_as = c:FSA
		}
		
	}
}

# ACW end, reconstruction start
acw_je_events.2 = {
	type = country_event
	event_image = {
		video = "unspecific_military_parade"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag_usa.dds"

	title = acw_je_events.2.t
	desc = acw_je_events.2.d
	flavor = acw_je_events.2.f

	duration = 3

	cooldown = { days = never_fire_again_modifier_time }

	trigger = {
		# triggered by je_acw_war completion
	}

	immediate = {
		# set_variable = {
		# 	name = reconstruction_var
		# 	value = 0
		# }
		if = {
			limit = { 
				has_variable = fsa_annex_var
			}
			remove_variable = acw_slave_nation_var
		}
		set_variable = {
			name = je_acw_wild_west_var
			value = 0
		}
		if = {
			limit = {
				OR = {
					has_variable = csa_will_spawn_var
					has_variable = fsa_annex_var
				}
			}
			every_scope_state = {
				limit = {
					is_incorporated = yes
					is_capital = no
					state_region = {
						is_homeland = cu:dixie
						NOT = { is_homeland = cu:yankee }
					}
					NOR = {
						is_state_region_dc = yes
						state_region = s:STATE_WEST_VIRGINIA
						is_state_region_maryland = yes
						is_state_region_delaware = yes
					}
				}
				set_state_type = unincorporated
				add_modifier = {
					name = unreconstructed_state
					years = 12
				}
				if = {
					limit = { has_building =  building_barrack }
					remove_building = building_barrack
				}
				if = {
					limit = { has_building = building_conscription_center }
					remove_building = building_conscription_center
				}
			}
		}
		else = {
			every_scope_state = {
				limit = {
					is_incorporated = yes
					is_capital = no
					state_region = {
						is_homeland = cu:yankee
						NOT = { is_homeland = cu:dixie }
					}
					NOR = {
						is_state_region_dc = yes
						state_region = s:STATE_WEST_VIRGINIA
						is_state_region_maryland = yes
						is_state_region_delaware = yes
					}
				}
				set_state_type = unincorporated
				add_modifier = {
					name = unreconstructed_state
					years = 12
				}
				if = {
					limit = { has_building =  building_barrack }
					remove_building = building_barrack
				}
				if = {
					limit = { has_building = building_conscription_center }
					remove_building = building_conscription_center
				}
			}
		}
	}

	#Historical Reconstruction
	option = {
		name = acw_je_events.2.a
		trigger = {
			OR = {
				has_variable = fsa_annex_var
				has_variable = csa_will_spawn_var
			}
			has_law_or_variant = law_type:law_slavery_banned
		}
		default_option = yes
		add_journal_entry = {
			type = je_acw_reconstruction
		}
		# add_journal_entry = { type = je_acw_reincorporate }		
		# if = {
		# 	limit = {
		# 		has_law_or_variant = law_type:law_slavery_banned
		# 	}
		# 	add_journal_entry = {
		# 		type = je_acw_equality
		# 	}
		# }
		# else = {
		# 	change_variable = {
		# 		name = reconstruction_var
		# 		add = 1
		# 	}
		# }
		add_journal_entry = {
			type = je_acw_wild_wild_west
		}
		je:je_acw_reconstruction ?= {
			add_modifier = {
				name = usfp_reconstruction_white_fatigue_modifier
				multiplier = var:reconstruction_var
			}
			add_modifier = {
				name = usfp_reconstruction_occupation_authority_modifier
				multiplier = 100
			}
		}
		ai_chance = {
			base = 1000
		}
		if = {
			limit = {
				has_variable = fsa_annex_var
			 }
			remove_variable = fsa_annex_var
			trigger_event = {
				id = usfp_reconstruction.16
				months = 1
			}
		}

		add_modifier = {
			name = usfp_post_acw_demobilization_modifier
			years = 25
			is_decaying = yes
		}
		if = {
			limit = { 
				any_scope_character = {
					has_template = usa_lincoln_character_template
					has_ideology = ideology:ideology_abolitionist
				}
			}
			random_scope_character = {
				limit = {
					has_template = usa_lincoln_character_template
					has_ideology = ideology:ideology_abolitionist
				}
				set_ideology = ideology:ideology_reformer
			}
		}
	}

	#Reconstruction of the North
	option = {
		name = acw_je_events.2.b
		trigger = {
			NOR = {
				has_variable = csa_will_spawn_var
				has_variable = fsa_annex_var
			}
		}
		#unfinished
		add_journal_entry = {
			type = je_acw_wild_wild_west
		}
		add_modifier = {
			name = usfp_post_acw_demobilization_modifier
			years = 25
			is_decaying = yes
		}
	}

	#CSA but we ban slavery now
	option = {
		name = acw_je_events.2.c
		trigger = {
			has_variable = csa_will_spawn_var
			NOT = {
				has_law_or_variant = law_type:law_slavery_banned
			}
		}
		activate_law = law_type:law_slavery_banned
		add_journal_entry = {
			type = je_acw_reconstruction
		}
		je:je_acw_reconstruction ?= {
			add_modifier = {
				name = usfp_reconstruction_white_fatigue_modifier
				multiplier = var:reconstruction_var
			}
			add_modifier = {
				name = usfp_reconstruction_occupation_authority_modifier
				multiplier = 100
			}
		}
		add_journal_entry = {
			type = je_acw_wild_wild_west
		}
		ai_chance = {
			base = 1000
		}
		add_modifier = {
			name = usfp_post_acw_demobilization_modifier
			years = 25
			is_decaying = yes
		}
	}
	#CSA but without banning slavery
	option = {
		name = acw_je_events.2.g
		trigger = {
			has_variable = csa_will_spawn_var
			NOT = {
				has_law_or_variant = law_type:law_slavery_banned
			}
		}
		add_journal_entry = {
			type = je_acw_wild_wild_west
		}
		ai_chance = {
			base = 0
		}
		add_radicals = {
			value = very_large_radicals
			culture = cu:afro_american
		}
		add_radicals = {
			value = very_large_radicals
			pop_type = slaves
		}
		add_modifier = {
			name = usfp_post_acw_demobilization_modifier
			years = 25
			is_decaying = yes
		}
		if = {
			limit = { 
				any_scope_character = {
					has_template = usa_lincoln_character_template
					has_ideology = ideology:ideology_abolitionist
				}
			}
			random_scope_character = {
				limit = {
					has_template = usa_lincoln_character_template
					has_ideology = ideology:ideology_abolitionist
				}
				set_ideology = ideology:ideology_usfp_abolitionist_radical
			}
		}
	}

	after = {
		recalculate_pop_ig_support = yes
		update_party_support = yes
	}
}
