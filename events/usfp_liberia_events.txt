	type = country_event
	placement = scope:liberia_scope.capital

	event_image = {
		video = "africa_construction_colony"
	}

	desc = usfp_liberia.1.d
	flavor = usfp_liberia.1.f
	dlc = dlc_usfp
	duration = 3

			save_scope_as = liberia_scope
		}

		}
	}

	#nah

		default_option = yes
		change_relations = {
			country = c:LIB
			value = 10
		}
		create_unidirectional_truce = {
			country = c:LIB
			months = 60
		}
		c:LIB = {
			post_notification = usfp_liberia_republic
		}
		ai_chance = {
			base = 100
		}
	}

	option = {

			make_independent = yes
		}

		}
		create_unidirectional_truce = {
			country = c:LIB
			months = 12
		}
		c:LIB = {
			post_notification = usfp_liberia_independent
		}
		ai_chance = {
			base = 0
		}

	option = {
		name = usfp_liberia.1.c
		default_option = yes
		highlighted_option = yes
		trigger = {
			ig:ig_landowners = {

			}
		}
		change_relations = {
			country = c:LIB
			value = 20
		}

			set_country_type = recognized
		}
		create_unidirectional_truce = {
			country = c:LIB
			months = 60
		}
		c:LIB = {
			post_notification = usfp_liberia_recognized
		}
		ai_chance = {
			base = 0
		}
		show_as_unavailable = { always = yes }
	}
}
