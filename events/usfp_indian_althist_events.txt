						set_ideology = ideology:ideology_republican_leader

#Cross-Border Raid
usfp_indian_althist.2 = {
	type = country_event
	placement = ROOT
	cooldown = {years = 5}
	event_image = {
		video = "southamerica_war_civilians"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_fire.dds"
	
	title = usfp_indian_althist.2.t
	desc = usfp_indian_althist.2.d
	flavor = usfp_indian_althist.2.f
	dlc = dlc_usfp
	duration = 3

	trigger = {
		any_scope_state = {
			region = sr:region_great_plains
			any_neighbouring_state = {
				owner = {
					is_country_type = decentralized
				}
			}
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				region = sr:region_great_plains
				any_neighbouring_state = {
					owner = {
						is_country_type = decentralized
					}
				}
			}
			save_scope_as = raid_state
			random_neighbouring_state = {
				limit = {
					owner = {
						is_country_type = decentralized
					}
				}
				save_scope_as = raiding_country_state
				owner = {
					save_scope_as = raiding_country
				}
			}
		}
	}

	#retaliate
	option = {
		name = usfp_indian_althist.2.a
		trigger = {
			usfp_has_plains_occupation_trigger = yes
		}
		highlighted_option = yes
		show_as_unavailable = { always = yes }
		scope:raid_state = {
			state_region = {
				add_devastation = 3
			}
			usfp_loyalize_settlers_in_state = {
				LOYALISTS = small_radicals
			}
		}
		scope:raiding_country_state = {
			state_region = {
				add_devastation = 5
			}
		}
		change_tension = {
			country = scope:raiding_country
			value = 20
		}
		ai_chance = {
			base = 50
		}
	}
	#fortify
	option = {
		name = usfp_indian_althist.2.b
		default_option = yes
		trigger = {
			usfp_has_plains_occupation_trigger = yes
		}
		highlighted_option = yes
		show_as_unavailable = { always = yes }
		scope:raid_state = {
			state_region = {
				add_devastation = 3
			}
		}
		add_treasury = -20000
		ai_chance = {
			base = 20
		}
	}
	#do nothing
	option = {
		name = usfp_indian_althist.2.c
		scope:raid_state = {
			state_region = {
				add_devastation = 3
			}
		}
		scope:raid_state = {
			add_radicals_in_state = {
				value = medium_radicals
			}
		}
		ai_chance = {
			base = 10
		}
	}

}
