namespace = USCWF

# Setup Event
USCWF.1 = {
	type = country_event
	hidden = yes

	immediate = {
		ruler = { set_variable = memento_mori_immunity }
		trigger_event = { id = USCWF.2 years = 8 }
		trigger_event = { id = USCWF.4 months = 7 }
		#Truces to reset AI consideration of where to begin colonies
		every_country = {
			limit = {
				is_country_type = decentralized
				country_is_in_north_america = yes
			}
			if = {
				limit = {
					c:USA = {
						is_ai = yes
					}
				 }
				 create_truce = {
					country = c:USA
					months = 1
				 }
			}
			if = {
				limit = {
					c:MEX = {
						is_ai = yes
					}
				 }
				 create_truce = {
					country = c:MEX
					months = 1
				 }
			}
		}
		c:COM = {
			if = {
				limit = {
					c:MEX = {
						is_ai = yes
					}
				 }
				 create_truce = {
					country = c:MEX
					months = 24
				 }
			}
		}
		ig:ig_petty_bourgeoisie = {
			leader = {
				set_variable = is_daniel_webster
			}
		}
	}
}

}

namespace = acw_je_events

# ACW startup
acw_je_events.1 = {
	type = country_event
	event_image = {
		video = "europenorthamerica_capitalists_meeting"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = {
		first_valid = {
			triggered_desc = {
				trigger = {
					has_variable = csa_will_spawn_var
				}
				desc = acw_je_events.1.t1
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = acw_je_events.1.t2
			}
		}
	}
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					has_variable = csa_will_spawn_var
				}
				desc = acw_je_events.1.d1
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = acw_je_events.1.d2
			}
		}
	}
	flavor = {
		first_valid = {
			triggered_desc = {
				trigger = {
					has_variable = csa_will_spawn_var
				}
				desc = acw_je_events.1.f1
			}
			triggered_desc = {
				trigger = {
					always = yes
				}
				desc = acw_je_events.1.f2
			}
		}
	}

	duration = 3

	trigger = {
		# triggered by je_acw_countdown completion
	}

	immediate = {
		if = {
			limit = {
				has_variable = csa_will_spawn_var
			}
			random_country = {
				limit = {	
					is_revolutionary = yes
					country_definition = root.country_definition
				}
				change_tag = CSA
				save_scope_as = csa_scope	
				
				c:USA = {
					every_scope_state = {
						limit = { OR = { is_slave_state = yes has_variable = former_slave_state } }
						set_state_owner = c:CSA
					}
				}
				s:STATE_DISTRICT_OF_COLUMBIA.region_state:CSA = { set_state_owner = c:USA } # LMAO
				s:STATE_MISSOURI.region_state:CSA = { set_state_owner = c:USA }
				s:STATE_KENTUCKY.region_state:CSA = { set_state_owner = c:USA }
				s:STATE_MARYLAND.region_state:CSA = { set_state_owner = c:USA }
				s:STATE_DELAWARE.region_state:CSA = { set_state_owner = c:USA }
				s:STATE_WEST_VIRGINIA.region_state:CSA = { set_state_owner = c:USA }
				s:STATE_NEBRASKA.region_state:CSA = { set_state_owner = c:USA }
				s:STATE_IOWA.region_state:CSA = { set_state_owner = c:USA }
				s:STATE_KANSAS.region_state:CSA = { set_state_owner = c:USA }
				s:STATE_COLORADO.region_state:CSA = { set_state_owner = c:USA }
				s:STATE_UTAH.region_state:CSA = { set_state_owner = c:USA }
				s:STATE_NEVADA.region_state:CSA = { set_state_owner = c:USA }
				s:STATE_CALIFORNIA.region_state:CSA = { set_state_owner = c:USA }

				# Re-enslave pops if the ACW happens after slavery has been abolished
				if = {
					limit = {
						c:USA = {
							has_variable = slavery_recently_abolished
						}						
					}
					
					every_scope_state = {
						enslave_discriminated_farm_workers = yes
					}					
				}
				
				if = { # Scared of Britain	
					limit = {
						NOT = { has_law = law_type:law_slave_trade }
					}
					activate_law = law_type:law_slave_trade	
				}						
				
				every_scope_state = {
					state_region = {
						add_claim = root
					}
				}
			}
		}
		else = {
			random_country = {
				limit = {	
					is_revolutionary = yes
					country_definition = root.country_definition
				}
				change_tag = FSA
				save_scope_as = fsa_scope
				
				c:USA = {
					every_scope_state = {
						limit = { NOR = { is_slave_state = yes has_variable = former_slave_state } }
						set_state_owner = c:FSA
					}
				}

				every_scope_state = {
					state_region = {
						add_claim = root
					}
				}				
				
				if = {
					limit = {
						NOT = { has_law = law_type:law_slavery_banned }
					}
					activate_law = law_type:law_slavery_banned
				}
			}
		}
		set_global_variable = {
			name = american_civil_war
			value = yes
		}
		random_diplomatic_play = {
			limit = {
				any_scope_play_involved = {
					OR = {
						this = scope:csa_scope
						this = scope:fsa_scope
					}
				}
			}
			remove_war_goal = {
				who = initiator
				war_goal = annex_country
			}
			add_war_goal = {
				holder = initiator
				type = revoke_all_claims
				target_country = target				
				primary_demand = yes
			}			
		}
		
		if = {
			limit = { exists = c:CSA }
			add_journal_entry = {
				type = je_acw_war
				target = scope:csa_scope
			}
			remove_modifier = missouri_compromise
			add_modifier = { name = a_house_divided_uscwf months = 48 }
			add_modifier = { name = civil_war_reshuffle_uscwf months = 48 }
			ig:ig_intelligentsia = { remove_ruling_interest_group = yes } 
			c:CSA = {
				set_variable = acw_slave_nation_var
				add_modifier = { name = a_house_divided_uscwf months = 48 }
			}
		}
		else_if = {
			limit = { exists = c:FSA }
			add_journal_entry = {
				type = je_acw_war
				target = scope:fsa_scope
			}
			set_variable = acw_slave_nation_var
			add_modifier = { name = a_house_divided_uscwf months = 48 }
			c:FSA = {
				add_modifier = { name = a_house_divided_uscwf months = 48 }
			}
		}
		set_variable = union_side_acw
	}

	option = {
		name = acw_je_events.1.a
		default_option = yes
	}
	option = {
		name = acw_je_events.1.b
		trigger = { is_player = yes }

		if = {
			limit = { exists = c:CSA }
			play_as = scope:csa_scope
		}
		else_if = {
			limit = { exists = c:FSA }
			play_as = scope:fsa_scope
		}
	}
}
