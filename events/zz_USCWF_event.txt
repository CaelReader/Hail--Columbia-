namespace = USCWF

# Setup Event
USCWF.1 = {
	type = country_event
	hidden = yes

	immediate = {
		ruler = { set_variable = memento_mori_immunity }
		trigger_event = { id = USCWF.2 years = 8 }
		trigger_event = { id = USCWF.4 months = 7 }
		#Truces to reset AI consideration of where to begin colonies
		every_country = {
			limit = {
				is_country_type = decentralized
				country_is_in_north_america = yes
				NOT = {
					this = c:CTF
				}
			}
			if = {
				limit = {
					c:USA = {
						is_ai = yes
					}
				 }
				 create_truce = {
					country = c:USA
					months = 1
				 }
			}
			if = {
				limit = {
					c:MEX = {
						is_ai = yes
					}
				 }
				 create_truce = {
					country = c:MEX
					months = 1
				 }
			}
		}
		c:COM = {
			if = {
				limit = {
					c:MEX = {
						is_ai = yes
					}
				 }
				 create_truce = {
					country = c:MEX
					months = 24
				 }
			}
		}
		ig:ig_petty_bourgeoisie = {
			leader = {
				set_variable = is_daniel_webster
				add_modifier = {
					name = usfp_great_triumvirate
				}
			}
		}
		ig:ig_intelligentsia = {
			leader = {
				add_modifier = {
					name = usfp_great_triumvirate
				}
			}
		}
		ig:ig_landowners = {
			leader = {
				add_modifier = {
					name = usfp_great_triumvirate
				}
			}
		}
	}
}
# Yeehaw Event
USCWF.2 = {
	type = country_event
	hidden = yes
	immediate = {
		if = {
			limit = {
				is_ai = yes
				c:TEX ?= { is_ai = yes }
			}
			create_diplomatic_pact = {
				country = c:TEX
				type = increase_relations
			}
			trigger_event = { id = USCWF.3 months = 15 }
		}
	}
}
