#Banana Wars start
usfp_banana_wars.1 = {
    type = country_event
    placement = scope:central_american_state_scope
    event_image = {
        video = "ep1_clandestine_meeting"
    }

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

    icon = "gfx/interface/icons/goods_icons/prestige_goods/gros_michel_banana_prestige.dds"

    title = usfp_banana_wars.1.t
    desc = usfp_banana_wars.1.d
    flavor = usfp_banana_wars.1.f
    dlc = dlc_usfp
    duration = 3
    cooldown = { days = never_fire_again_modifier_time } #Once

    trigger = {
        #JE Start
    }

    immediate = {
        ordered_state_in_caribbean_coast = {
            position = 0
            order_by = gdp
            save_scope_as = central_american_state_scope
        }
    }

    #accept
    option = {
        name = usfp_banana_wars.1.a
        default_option = yes
        if = {
            limit = {
                any_scope_pop = {
                    culture = cu:central_american
                }
            }
            add_radicals = {
                value = large_radicals
                culture = cu:central_american
            }
        }
        if = {
            limit = {
                any_scope_pop = {
                    culture = cu:colombian
                }
            }
            add_radicals = {
                value = large_radicals
                culture = cu:colombian
            }
        }
        if = {
            limit = {
                any_scope_pop = {
                    culture = cu:north_andean
                }
            }
            add_radicals = {
                value = large_radicals
                culture = cu:north_andean
            }
        }
        show_as_tooltip = {
            add_journal_entry = {
                type = usfp_je_banana_wars
            }
        }
        every_country_in_caribbean_coast = {
            limit = {
                NOT = {
                    this = root
                }
            }
            trigger_event = usfp_banana_wars.4
        }
        ai_chance = {
            base = 100
        }
    }

    #reject
    option = {
        name = usfp_banana_wars.1.b
        ig:ig_industralists ?= {
            add_modifier = {
                name = usfp_rejected_banana_wars_ig_modifier
                years = 4
            }
        }
        add_radicals = {
            pop_type = capitalists
            value = small_radicals
        }
        custom_tooltip = {
            text = close_banana_wars
            set_variable = rejected_banana_wars
        }
        ai_chance = {
            base = 0
        }
    }
}

usfp_banana_wars.3 = {
    type = country_event
    placement = scope:central_american_state_scope

    desc = usfp_banana_wars.3.d
    flavor = usfp_banana_wars.3.f
    dlc = dlc_usfp

    cooldown = { days = never_fire_again_modifier_time } #Once

    trigger = {
        any_scope_state = {
            is_in_geographic_region = geographic_region_caribbean_coast
            is_incorporated = no
            is_treaty_port = no
        }
    }

    immediate = {
        ordered_scope_state = {
            limit = {
                is_in_geographic_region = geographic_region_caribbean_coast
                is_incorporated = no
                is_treaty_port = no
            }
            order_by = gdp
            position = 0
            save_scope_as = central_american_state_scope
        }
        if = {
            limit = {
                has_company = company_type:company_united_fruit
            }
            company:company_united_fruit = {
                save_scope_as = united_fruit_scope
            }
        }
    }

    #do not incorporate
    option = {
        name = usfp_banana_wars.3.a
        default_option = yes
        ig:ig_industrialists ?= {
            add_modifier = {
                name = usfp_protected_ig_interests_modifier
                years = 4
            }
        }
        every_scope_state = {
            limit = {
                is_in_geographic_region = geographic_region_caribbean_coast
                is_incorporated = no
                is_treaty_port = no
            }
            add_modifier = usfp_banana_wars_colony_modifier
            add_radicals_in_state = {
                value = large_radicals
            }
        }
        ai_chance = {
            base = 100
        }
    }

    #actually we want to turn these into states
    option = {
        name = usfp_banana_wars.3.b
        usfp_manifest_more_destiny_low_effect = yes
        every_scope_state = {
            limit = {
                is_in_geographic_region = geographic_region_caribbean_coast
                is_incorporated = no
                is_treaty_port = no
            }
            add_modifier = {
                name = usfp_banana_wars_capitalist_interference_modifier
                years = 5
            }
            add_radicals_in_state = {
                value = medium_radicals
            }
        }
        ig:ig_industrialists ?= {
            add_modifier = {
                name = usfp_industrialists_displeased_banana_wars_ig_modifier
                years = 4
            }
        }
        ai_chance = {
            base = 0
        }
    }

    #United Fruit Company
    option = {
        name = usfp_banana_wars.3.c
        highlighted_option = yes
        trigger = {
            has_company = company_type:company_united_fruit
        }
        ig:ig_industrialists ?= {
            add_modifier = {
                name = usfp_industrialists_banana_empire_ig_modifier
                years = 6
            }
        }
        custom_tooltip = {
            text = create_united_fruit_country
            create_country = {
                tag = usfp_UFC
                origin = root
                state = scope:central_american_state_scope
            }
            scope:united_fruit_scope = {
                add_owned_country = c:usfp_UFC
            }
            every_scope_state = {
                limit = {
                    is_in_geographic_region = geographic_region_caribbean_coast
                    is_incorporated = no
                    is_treaty_port = no
                }
                set_state_owner = c:usfp_UFC
            }
            create_diplomatic_pact = {
                country = c:usfp_UFC
                type = chartered_company
            }
            c:usfp_UFC ?= {
                add_liberty_desire = -100
                activate_law = law_type:law_land_based_taxation
                activate_law = law_type:law_appointed_bureaucrats
                activate_law = law_type:law_colonial_administration
                activate_law = law_type:law_extraction_economy
                activate_law = law_type:law_oligarchy
                activate_law = law_type:law_dedicated_police
                activate_law = law_type:law_no_schools
                activate_law = law_type:law_no_workers_rights
                activate_law = law_type:law_no_health_system
                activate_law = law_type:law_no_home_affairs
                activate_law = law_type:law_no_migration_controls
                activate_law = law_type:law_no_colonial_affairs
                activate_law = law_type:law_censorship
                activate_law = law_type:law_child_labor_allowed
                activate_law = law_type:law_no_social_security
                activate_law = law_type:law_combination_acts
                activate_law = law_type:law_commercialized_agriculture
                activate_law = law_type:law_free_trade
                ig:ig_industrialists ?= {
                    if = {
                        limit = {
                            has_ideology = ideology:ideology_laissez_faire
                        }
                        remove_ideology = ideology_laissez_faire
                        add_ideology = ideology_colonialist
                    }
                }
                set_ruling_interest_groups = { ig_industrialists ig_armed_forces }
            }
        }
        ai_chance = {
            base = 100
        }
    }
}

                OR = {
                    movement_is_socialist = yes
                    is_political_movement_type = movement_labor

                    political_movement_radicalism >= 0.25

        }

                        movement_is_socialist = yes
                        is_political_movement_type = movement_labor

                        political_movement_support >= 0.10
                        political_movement_radicalism >= 0.25
                    }

                OR = {
                    is_subject_type = subject_type_colony
                    is_subject_type = subject_type_chartered_company

        name = usfp_banana_wars.7.a
        default_option = yes

            add_radicals = {
                value = medium_radicals
