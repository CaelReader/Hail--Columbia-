## ANTI-SLAVERY
#Human Rights tech doubles abolition support
usfp_movement_anti_slavery_pop_support_mult = {
	if = {
		limit = {
			is_pop_type = slaves
			state = {
				OR = {
					has_modifier = usfp_deep_south_modifier
					has_modifier = usfp_upper_south_modifier
				}
			}
		}
		multiply = {
			value = 0.1
			desc = "usfp_slave_power_modifier"
		}
	}
	else_if = {
		limit = {
			owner ?= {
				has_technology_researched = human_rights
			}
		}
		multiply = {
			value = 2.0
			desc = "human_rights"	
		}
	}
}

}

usfp_movement_anti_slavery_additional_radicalism_add = {
	#By the time you have Human Rights tech slavery should really be getting abolished
	if = {
		limit = {
			owner ?= {
				has_technology_researched = human_rights
			}
		}
		add = {
			value = 0.2
			desc = "human_rights"
		}
	}
	#a powerful abolitionist movement should go for it
	if = {
		limit = {
			political_movement_support > 0.33
		}
		add = {
			value = political_movement_support
			desc = "[concept_political_movement_support]"
		}
	}
	if = {
		limit = {
			owner ?= {
				has_journal_entry = je_acw_countdown
				NOT = {
					has_journal_entry = je_usfp_a_house_divided_missouri_compromise
				}
			}
		}
		add = {
			value = {
				owner ?= {
					every_scope_state = {
						limit = {
							is_incorporated = yes
							is_slave_state = yes
							NOR = {
								state_region = s:STATE_DISTRICT_OF_COLUMBIA
								state_region = s:STATE_WEST_VIRGINIA
							}
						}
						add = 0.025
					}
				}
				owner ?= {
					every_scope_state = {
						limit = {
							is_incorporated = yes
							is_slave_state = no
							NOR = {
								state_region = s:STATE_DISTRICT_OF_COLUMBIA
								state_region = s:STATE_WEST_VIRGINIA
							}
						}
						subtract = 0.025
					}
				}
			}
			desc = "slave_free_state_balance"
		}
	}
}

## PRO-SLAVERY
usfp_movement_pro_slavery_secession_state_weight_mult = {
	#USFP: Deep South and Upper South more inclined to join
	if = {
		limit = {
			is_usfp_active = yes
		}
		#USFP: Deep South and Upper South more inclined to join
		if = {
			limit = {
				is_usfp_active = yes
				has_modifier = usfp_deep_south_modifier
			}
			value = 5.0
		}
		else_if = {
			limit = {
				is_usfp_active = yes
				has_modifier = usfp_upper_south_modifier
			}
			value = 2.0
		}
		#USFP: West Virginia goes with real Virginia
		if = {
			limit = {
				is_usfp_active = yes
				state_region = s:STATE_WEST_VIRGINIA
			}
			value = 10.0
		}
		#USFP: Missouri is less likely
		if = {
			limit = {
				is_usfp_active = yes 
				state_region = s:STATE_MISSOURI
			}
			value = 0.5
		}	
		else = {
			value = 1	# Mult no change
		}
	}
}
#Powerful IG with anti-slavery leader spooks the slavers
usfp_movement_pro_slavery_additional_radicalism_add = {
	if = {
		limit = {
			is_usfp_active = yes
			owner ?= {
				usfp_country_is_american = yes
				any_interest_group = {
					is_powerful = yes
					leader = {
						character_is_anti_slavery = yes
					}
				}
			}
		}
		add = {
			value = 0.3
			desc = "powerful_abolitionist"
		}
	}
	if = {
		limit = {
			owner ?= {
				has_journal_entry = je_acw_countdown
				NOT = {
					has_journal_entry = je_usfp_a_house_divided_missouri_compromise
				}
			}
		}
		add = {
			value = {
				owner ?= {
					every_scope_state = {
						limit = {
							is_incorporated = yes
							is_slave_state = yes
							NOR = {
								state_region = s:STATE_DISTRICT_OF_COLUMBIA
								state_region = s:STATE_WEST_VIRGINIA
							}
						}
						subtract = 0.025
					}
				}
				owner ?= {
					every_scope_state = {
						limit = {
							is_incorporated = yes
							is_slave_state = no
							NOR = {
								state_region = s:STATE_DISTRICT_OF_COLUMBIA
								state_region = s:STATE_WEST_VIRGINIA
							}
						}
						add = 0.025
					}
				}
			}
			desc = "slave_free_state_balance"
		}
	}
}
