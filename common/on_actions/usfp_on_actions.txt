#this tries to prevent native uprisings in the far north where the impassable terrain would cause problems
usfp_arctic_isolation = {
	effect = {
		if = {
			limit = {
				OR = {
					s:STATE_ALASKA = {
						root_has_state_here = yes
					}
					s:STATE_YUKON_TERRITORY = {
						root_has_state_here = yes
					}
					s:STATE_NORTHWEST_TERRITORIES = {
						root_has_state_here = yes
					}
					s:STATE_NUNAVUT = {
						root_has_state_here = yes
					}
					s:STATE_QUEBEC = {
						root_has_state_here = yes
					}
				} 
			}
			every_scope_state = {
				limit = {
					OR = {
						state_region = s:STATE_ALASKA
						state_region = s:STATE_YUKON_TERRITORY
						state_region = s:STATE_NORTHWEST_TERRITORIES
						state_region = s:STATE_NUNAVUT
						state_region = s:STATE_QUEBEC
					}
				}
				every_neighbouring_state = {
					limit = {
						owner = {
							is_country_type = decentralized
							OR = {
								country_has_primary_culture = cu:athabaskan
								country_has_primary_culture = cu:inuit
							}
						}
						OR = {
							state_region = s:STATE_ALASKA
							state_region = s:STATE_YUKON_TERRITORY
							state_region = s:STATE_NORTHWEST_TERRITORIES
							state_region = s:STATE_NUNAVUT
							state_region = s:STATE_QUEBEC
						}
					}
					owner = {
						set_tension = {
							country = root
							value = 0
						}
					}
				}
			}
		}
	}
}

usfp_new_president = {
	effect = {
		owner = {
			trigger_event = usfp_usa.21
		}
		#manually force ai strats on a new president
		if = {
			limit = {
				owner = { is_ai = yes}
			 }
			if = {
				limit = {
					usfp_manifest_destiny_pro_expansion_trigger = yes
					owner = {
						has_journal_entry = usfp_manifest_destiny_main
						NOT = {
							has_global_variable = american_frontier_closed
						}
					} 
				}
				owner = {
					set_strategy = ai_strategy_manifest_destiny
				}
			} else_if = {
				limit = {
					usfp_manifest_destiny_pro_expansion_trigger = yes
					owner = {
						has_journal_entry = usfp_manifest_destiny_main
						has_global_variable = american_frontier_closed
					} 
				}
				owner = {
					set_strategy = ai_strategy_pacific_empire
				}
			} else_if = {
				limit = {
					usfp_manifest_destiny_anti_expansion_trigger = yes
				}
				owner = {
					set_strategy = ai_strategy_monroe_doctrine
				}
			}
			if = {
				limit = {
					interest_group = {
						is_interest_group_type = ig_rural_folk
					}
				 }
				 owner = {
					set_strategy = ai_strategy_agricultural_expansion
				 }
			}
			else_if = {
				limit = {
					interest_group = {
						is_interest_group_type = ig_landowners
					}
				 }
				 owner = {
					set_strategy = ai_strategy_plantation_economy
				 }
			}
			else_if = {
				limit = {
					OR = {
						interest_group = {
							is_interest_group_type = ig_petty_bourgeoisie
							is_interest_group_type = ig_trade_unions
							is_interest_group_type = ig_intelligentsia
							is_interest_group_type = ig_industrialists
						}
						owner = {
							ig:ig_industrialists = {
								is_in_government = yes
							}
						}
					}
				 }
				 owner = {
					set_strategy = ai_strategy_industrial_expansion
				 }
			}
			if = {
				limit = { 
					OR = {
						has_ideology = ideology:ideology_usfp_progressive
						has_ideology = ideology:ideology_reformer
					}
				}
				owner = {
					set_strategy = ai_strategy_progressive_agenda
				}
			}
			if = {
				limit = {
					usfp_anti_slavery_leader_trigger = yes
					owner = {
						NOT = {
							has_law_or_variant = law_type:law_slavery_banned
						}
					}
				 }
				 owner = {
					set_strategy = ai_strategy_anti_slavery_agenda
				 }
			}

		}
		

	}
}

USCWF_precedent_on_action = {
	effect = {
		if = {
			limit = {
				c:USA ?= this
				has_journal_entry = je_usfp_american_constitution
			}
			ruler = {
				save_scope_as = ruler_scope
				if = {
					limit = {
						NOT = { has_variable = var_elected_once }
					}
					set_variable = var_elected_once
					debug_log = "USFP: President has now been elected once"
				}
				else = {
					if = {
						limit = {
							usfp_president_might_go_for_third_term = no
							NOT = {
								has_variable = president_broke_precedent
							}
						 }
						 debug_log = "USFP: Washington Precedent triggered"
						 usfp_washington_precedent_effect = yes 
					}
					else_if = {
						limit = {
							usfp_president_might_go_for_third_term = yes
						}
						debug_log = "USFP: President might break Washington Precedent!"
						set_variable = president_might_break_precedent
					}	
				}
			}
		}
	}
}

on_law_enactment_started = {
	on_actions = {
		USCWF_CSA_betrayed_on_action
	}
}

USCWF_CSA_betrayed_on_action = {
	effect = {
		if = {
			limit = {
				c:USA ?= this
				is_enacting_law = law_type:law_slavery_banned
				any_interest_group = {
					has_modifier = ig_slavers_happy
				}
			}
			every_interest_group = {
				limit = {
					OR = {
						has_ideology = ideology:ideology_pro_slavery
						leader = { has_ideology = ideology:ideology_slaver }
					}
				}
				remove_modifier = ig_slavers_happy
				add_modifier = {
					name = ig_slavers_mad
					days = very_long_modifier_time
					multiplier = 6
				}
			}
		}
	}
}

on_political_movement_formed = {
	on_actions = {
		usfp_political_movement_ideologies
	}
}

usfp_political_movement_ideologies = {
	effect = {
		if = {
			limit = {
				is_political_movement_type = movement_cultural_minority 
				culture = cu:afro_american
				owner = {
					usfp_country_is_american = yes
				}
			}
			set_core_ideology = ideology_usfp_afro_american_movement
		}
		if = {
			limit = {
				is_political_movement_type = movement_cultural_minority 
				culture = {
					has_discrimination_trait_group = heritage_group_indigenous_american
				}
				owner = {
					is_in_geographic_region = geographic_region_north_america
				}
			}
			set_core_ideology = ideology_usfp_american_indian_movement
		}
	}
}

on_diplomatic_play_started = {
	on_actions = {
		usfp_manifest_destiny_decay_reset
	}
}

usfp_manifest_destiny_decay_reset = {
	effect = {
		scope:initiator = {
			if = {
				limit = {
					has_journal_entry = usfp_manifest_destiny_main
					has_variable = manifest_destiny_decaying_warpath
				}
				if = {
					limit = { 
						has_variable = manifest_destiny_decay_clock
					}
					remove_variable = manifest_destiny_decay_clock
				}
				set_variable = {
					name = manifest_destiny_decay_clock
					years = 3
				}
			}
		}
	}
}

on_war_end = {
	on_actions = {
		usfp_on_war_end_action
	}
}

usfp_on_war_end_action = {
	effect = {
		scope:actor = {
			if = {
				limit = { 
					c:USA ?= this
				}
				set_variable = {
					name = red_summer_possible
					years = 1
				}
			}
		}
		scope:target ?= {
			if = {
				limit = { 
					c:USA ?= this
				}
				set_variable = {
					name = red_summer_possible
					years = 1
				}
			}
		}
	}
}

on_government_reformed = {
	on_actions = {
		usfp_spoils_system_action
	}
}

usfp_spoils_system_action = {
	effect = {
		if = {
			limit = {
				active_law:lawgroup_bureaucracy ?= {
					has_amendment = amendment_type:amendment_usfp_spoils_system
				}
				NOT = {
					has_modifier = usfp_spoils_system_decaying_modifier
				}
			 }
			 add_modifier = {
				name = usfp_spoils_system_decaying_modifier
				years = 2
				is_decaying = yes
			 }
			 change_variable = {
				name = usfp_spoils_system_abolish_var
				add = 1
			 }
		}
	}
}
