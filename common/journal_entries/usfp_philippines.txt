je_usfp_philippine_tutelage = {
    icon = "gfx/interface/icons/event_icons/waving_flags_usa_philippines.dds"

    group =  je_group_us_expansion_global

    should_be_involved = {
        OR = {
            c:USA ?= this
            c:PHI ?= this
        }
    }

    should_show_when_not_involved = {
        OR = {
            c:USA ?= this
            c:PHI ?= this
        }
    }

    on_monthly_pulse = {
        random_events = {

        }
    }

    immediate = {
        scope:journal_entry = {
            add_com_dlc_icon = {
                dlc = dlc_usfp
            }
        }
    }

    immediate_all_involved = {
        c:PHI = {
            save_scope_as = philippines_scope
        }
        c:USA = {
            save_scope_as = usa_scope
        }
    }

    is_shown_when_inactive = {
        exists = c:PHI
        exists = c:USA
        c:PHI ?= {
            is_subject = yes
        }
        NOT = {
            c:PHI ?= {
                has_law_or_variant = c:USA.active_law:lawgroup_governance_principles.type
                trigger_if = {
                    limit = {
                        c:USA = {
                            NOT = {
                                has_law_or_variant = law_type:law_universal_suffrage
                            }
                        }
                    }
                    has_law_or_variant = c:USA.active_law:lawgroup_distribution_of_power.type
                } trigger_else = {
                    country_has_voting_franchise = yes
                }
                OR = {
                    has_law_or_variant = law_type:law_public_health_insurance
                    has_law_or_variant = law_type:law_private_health_insurance
                }
                OR = {
                    has_law_or_variant = law_type:law_public_schools
                    has_law_or_variant = law_type:law_private_schools
                }
                NOR = {
                    has_law_or_variant = law_type:law_serfdom
                    has_law_or_variant = law_type:law_tenant_farmers
                }
                has_law_or_variant = law_type:law_compulsory_primary_school
                NOT = {
                    has_law_or_variant = law_type:law_state_religion
                }
                has_law_or_variant = law_type:law_slavery_banned
            }
        }
    }

    possible = {
        c:USA = {
            has_technology_researched = civilizing_mission
        }
        c:PHI ?= {
            is_subject_of = c:USA
        }
    }

    scripted_button = usfp_philippine_insular_government_button

    complete = {
        c:PHI ?= {
            legitimacy >= 50
            has_law_or_variant = c:USA.active_law:lawgroup_governance_principles.type
            trigger_if = {
                limit = {
                    c:USA = {
                        NOT = {
                            has_law_or_variant = law_type:law_universal_suffrage
                        }
                    }
                }
                has_law_or_variant = c:USA.active_law:lawgroup_distribution_of_power.type
            } trigger_else = {
                country_has_voting_franchise = yes
            }
            OR = {
                has_law_or_variant = law_type:law_public_health_insurance
                has_law_or_variant = law_type:law_private_health_insurance
            }
            OR = {
                has_law_or_variant = law_type:law_public_schools
                has_law_or_variant = law_type:law_private_schools
            }
            NOR = {
                has_law_or_variant = law_type:law_serfdom
                has_law_or_variant = law_type:law_tenant_farmers
            }
            has_law_or_variant = law_type:law_compulsory_primary_school
            NOT = {
                has_law_or_variant = law_type:law_state_religion
            }
            has_law_or_variant = law_type:law_slavery_banned
            NOT = {
                any_political_movement = {
                    is_insurrectionary = yes
                }
            }
        }
    }

    on_complete = {

    }

    fail = {
        NOT = {
            c:PHI ?= {
                is_subject_of = c:USA
            }
        }
    }

    on_fail = {
        c:PHI ?= {
            if = {
                limit = {
                    has_modifier = usfp_philippines_military_government_modifier
                }
                remove_modifier = usfp_philippines_military_government_modifier
            }
        }
    }

    invalid = {
        NOT = {
            exists = c:USA
            exists = c:PHI
        }
        custom_tooltip = {
            text = usa_rejected_philippine_tutelage
            c:USA = {
                has_variable = rejected_philippines_tutelage
            }
        }

    }

    on_invalid = {
        c:PHI ?= {
            if = {
                limit = {
                    has_modifier = usfp_philippines_military_government_modifier
                }
                remove_modifier = usfp_philippines_military_government_modifier
            }
        }
    }


    weight = 200
    should_be_pinned_by_default_uninvolved_or_context = yes
}
