#Missouri Compromise
je_usfp_a_house_divided_missouri_compromise = {
	icon = "gfx/interface/icons/generic_icons/truce.dds"

	group = je_group_usa_american_civil_war

	should_be_pinned_by_default = yes

	modifiers_while_active = {
		usfp_missouri_compromise_modifier
	}

	immediate = {
		s:STATE_MISSOURI.region_state:USA = {
			save_scope_as = missouri_scope
		}
	}

	fail = {
		OR = {
			NOT = {
				root = {
					has_state_in_state_region = STATE_MISSOURI
				}
			}
			has_law = law_type:law_slavery_banned
			has_law = law_type:law_slave_trade
			custom_tooltip = {
				text = usfp_missouri_compromise_broken
				has_variable = missouri_compromise_broken_var
			}
			
		}

	}

	on_fail = {
		usfp_break_compromises_effect = yes
	}

	invalid = {
		NOT = {
			has_journal_entry = je_acw_countdown
		}
	}

	on_invalid = {
		usfp_break_compromises_effect = yes
	}
}

#Gag Rule: separate JE to be transparent with failure conditions
je_usfp_a_house_divided_gag_rule = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_usa_american_civil_war

	should_be_pinned_by_default = yes

	modifiers_while_active = {
		usfp_gag_rule_modifier
	}

	scripted_button = usfp_end_gag_rule_button
	scripted_button = usfp_break_gag_rule_button

	status_desc = {
		first_valid = {
			triggered_desc = { 
				desc = je_usfp_a_house_divided_gag_rule_status
			}
		}
	}

	fail = {
		ig:ig_landowners = {
			is_in_government = no
		}
		government_legitimacy >= 51
	}

	invalid = {
		OR = {
			NOT = {
				has_journal_entry = je_acw_countdown
			}
			has_variable = removed_gag_rule
		}
	}
}

#Bleeding Kansas: after the Missouri Compromise is broken
je_usfp_a_house_divided_bleeding_kansas = {
	icon = "gfx/interface/icons/generic_icons/secession_culture_ongoing.dds"

	group = je_group_usa_american_civil_war

	should_be_pinned_by_default = yes

	scripted_button = usfp_support_free_kansas_button
	scripted_button = usfp_support_slave_kansas_button
	scripted_button = usfp_enforce_kansas_constitution_button
	scripted_progress_bar = usfp_bleeding_kansas_progress_bar
	immediate = {
		s:STATE_KANSAS.region_state:USA = {
			save_scope_as = kansas_scope
		}
		cu:dixie = {
			save_scope_as = dixie_culture
		}
		cu:yankee = {
			save_scope_as = yankee_culture
		}
		ideology:ideology_abolitionist = {
			save_scope_as = abolitionist_scope
		}
		ideology:ideology_slaver = {
			save_scope_as = slaver_scope
		}
	}

	on_monthly_pulse = {
		random_events = {
			200 = 0
			100 = historical_agitators.5 # John Brown spawn
			20 = acw_events.6 # The Caning of Charles Sumner
			100 = usfp_bleeding_kansas.1 #New England Emigrant Aid Society
			20 = usfp_bleeding_kansas.2 #Border Ruffians
			20 = usfp_bleeding_kansas.3 #Ballot Violence
			50 = usfp_bleeding_kansas.4 #Sacking of Lawrence
			50 = usfp_bleeding_kansas.5 #Proslavery Men Massacred
			50 = usfp_bleeding_kansas.6 #Harper's Ferry (historical)
			1 = usfp_bleeding_kansas.7 #Harper's Ferry (successful)
			20 = usfp_bleeding_kansas.8 #Competing Constitutions
		}
	}

	timeout = 2190

	on_timeout = {
		if = {
			limit = {
				scope:journal_entry = {
					"scripted_bar_progress(usfp_bleeding_kansas_progress_bar)" <= 45
				}
			}
			#Slave Kansas
			trigger_event = {
				id = usfp_bleeding_kansas.10
				popup = yes
			}
			show_as_tooltip = {
				scope:kansas_scope = {
					add_loyalists_in_state = {
						value = large_radicals
						culture = cu:dixie
					}
					enslave_discriminated_farm_workers = yes
					if = {
						limit = { 
							is_incorporated = no
						}
						set_state_type = incorporated
					}
					if = {
						limit = { 
							cu:dixie = {
								NOT = {
									has_homeland = prev.state_region
								}
							}
						}
						state_region = {
							add_homeland = cu:dixie
						}
					}
					if = {
						limit = { 
							exists = c:CSA
						}
						set_state_owner = c:CSA
					}
				}
				if = {
					limit = {
						exists = c:CSA
						has_state_in_state_region = STATE_MISSOURI
					}
					s:STATE_MISSOURI.region_state:USA = {
						set_state_owner = c:CSA
					}
				}
			}
		} else = {
			#Free Kansas
			trigger_event = {
				id = usfp_bleeding_kansas.9
				popup = yes
			}
			show_as_tooltip = {
				scope:kansas_scope = {
				add_loyalists_in_state = {
					value = large_radicals
					culture = cu:yankee
				}
				if = {
					limit = { 
						any_scope_pop = {
							is_pop_type = slaves
						}
					}
					liberate_slaves = yes
				}
				if = {
					limit = { 
						is_incorporated = no
					}
					set_state_type = incorporated
				}
				if = {
					limit = { 
						cu:yankee = {
							NOT = {
								has_homeland = prev.state_region
							}
						}
					}
					state_region = {
						add_homeland = cu:yankee
					}
				}
				}
			}
		}
	}

	complete = {
		OR = {
			custom_tooltip = {
				text = bleeding_kansas_0_progress
				scope:journal_entry = {
					"scripted_bar_progress(usfp_bleeding_kansas_progress_bar)" <= 0
				}
			}
			custom_tooltip = {
				text = bleeding_kansas_100_progress
				scope:journal_entry = {
					"scripted_bar_progress(usfp_bleeding_kansas_progress_bar)" >= 100
				}
			}
			custom_tooltip = {
				text = acw_will_happen_tt
				has_global_variable = american_civil_war
			}
		}
	}

	on_complete = {
		if = {
			limit = {
				scope:journal_entry = {
					"scripted_bar_progress(usfp_bleeding_kansas_progress_bar)" <= 45
				}
			}
			#Slave Kansas
			trigger_event = {
				id = usfp_bleeding_kansas.10
				popup = yes
			}
			show_as_tooltip = {
				scope:kansas_scope = {
					add_loyalists_in_state = {
						value = large_radicals
						culture = cu:dixie
					}
					enslave_discriminated_farm_workers = yes
					if = {
						limit = { 
							is_incorporated = no
						}
						set_state_type = incorporated
					}
					if = {
						limit = { 
							cu:dixie = {
								NOT = {
									has_homeland = prev.state_region
								}
							}
						}
						state_region = {
							add_homeland = cu:dixie
						}
					}
					if = {
						limit = { 
							exists = c:CSA
						}
						set_state_owner = c:CSA
					}
				}
				if = {
					limit = {
						exists = c:CSA
						has_state_in_state_region = STATE_MISSOURI
					}
					s:STATE_MISSOURI.region_state:USA = {
						set_state_owner = c:CSA
					}
				}
			}
		} else = {
			#Free Kansas
			trigger_event = {
				id = usfp_bleeding_kansas.9
				popup = yes
			}
			show_as_tooltip = {
				scope:kansas_scope = {
				add_loyalists_in_state = {
					value = large_radicals
					culture = cu:yankee
				}
				if = {
					limit = { 
						any_scope_pop = {
							is_pop_type = slaves
						}
					}
					liberate_slaves = yes
				}
				if = {
					limit = { 
						is_incorporated = no
					}
					set_state_type = incorporated
				}
				if = {
					limit = { 
						cu:yankee = {
							NOT = {
								has_homeland = prev.state_region
							}
						}
					}
					state_region = {
						add_homeland = cu:yankee
					}
				}
				}
			}
		}
	}

	invalid = {
		NOT = {
			any_scope_state = {
				state_region = s:STATE_KANSAS
			}
		}
		NOT = {
			has_law = law_type:law_legacy_slavery
		}
	}
	on_invalid = {
		hidden_effect = {
			usfp_cleanup_bleeding_kansas_variables = yes
		}
	}

}

	should_be_pinned_by_default = yes

	modifiers_while_active = {
		usfp_black_suffrage_denied_modifier
	}

	immediate = {
		cu:afro_american = {
			save_scope_as = afro_american_culture
		}
	}

	complete = {
		OR = {
			has_law = law_type:law_cultural_exclusion
			has_law = law_type:law_multicultural

	}

	on_complete = {
		add_loyalists = {
			value = medium_radicals
			culture = cu:afro_american
		}
		remove_modifier = usfp_black_suffrage_denied_modifier
	}



	invalid = {
		OR = {
			country_has_voting_franchise = no
			has_law = law_type:law_council_republic
		}
	}

}

je_usfp_underground_railroad = {
	icon = "gfx/interface/icons/invention_icons/railways.dds"

	group = je_group_usa_american_civil_war

	should_be_pinned_by_default = yes

	immediate = {
		cu:afro_american = {
			save_scope_as = afro_american_culture
		}
		set_variable = {
			name = je_underground_railroad_northern_escapees
			value = 0
		}
		set_variable = {
			name = je_underground_railroad_canada_escapees
			value = 0
		}
		set_variable = {
			name = je_underground_railroad_southern_escapees
			value = 0
		}
		c:MEX = {
			trigger_event = usfp_a_house_divided.21
		}
		c:QUE = {
			trigger_event = usfp_a_house_divided.21
		}
		c:GBR = {
			trigger_event = usfp_a_house_divided.21
		}
		c:HBC = {
			trigger_event = usfp_a_house_divided.21
		}
		c:ONT = {
			trigger_event = usfp_a_house_divided.21
		}
		c:HAI = {
			trigger_event = usfp_a_house_divided.21
		}
	}

	scripted_button = usfp_underground_railroad_support_button
	scripted_button = usfp_underground_railroad_suppress_button
	scripted_button = usfp_underground_railroad_end_interference_button
	scripted_button = usfp_underground_railroad_arm_slaves_button

	on_monthly_pulse = {
		effect = {
			#between 100k and 500k slaves escaped historically, so this is targeting 500k as an upper bound. Actual amount of escapees will vary widely due to the different sizes of pops targeted
			random_list = {
				400 = {
					trigger = {
						NOR = {
							je:je_usfp_underground_railroad = {
								has_modifier = usfp_underground_railroad_suppressing_modifier
							}
							je:je_acw_countdown ?= {
								has_modifier = usfp_fugitive_slave_law
							}
						}
					}
					usfp_underground_railroad_escapes_to_north_effect = yes #escapees from three different states
					usfp_underground_railroad_escapes_to_north_effect = yes
					usfp_underground_railroad_escapes_to_north_effect = yes
				}
				100 = { 
					usfp_underground_railroad_escapes_to_canada_effect = yes #escapees from two different states
					usfp_underground_railroad_escapes_to_canada_effect = yes  
				}
				10 = { 
					usfp_underground_railroad_escapes_southern_effect = yes 
				}
				600 = {
					trigger = {
						je:je_usfp_underground_railroad = {
							has_modifier = usfp_underground_railroad_supporting_modifier
						}
					}
					usfp_underground_railroad_escapes_to_north_effect = yes
					usfp_underground_railroad_escapes_to_north_effect = yes
					usfp_underground_railroad_escapes_to_north_effect = yes
					usfp_underground_railroad_escapes_to_north_effect = yes
				}
				600 = {
					trigger = {
						exists = c:CSA
					}
					usfp_underground_railroad_escapes_to_north_effect = yes
					usfp_underground_railroad_escapes_to_north_effect = yes
					usfp_underground_railroad_escapes_to_north_effect = yes
					usfp_underground_railroad_escapes_to_north_effect = yes
				}
				400 = {
					trigger = {
						je:je_acw_countdown ?= {
							has_modifier = usfp_fugitive_slave_law
						}
					}
					usfp_underground_railroad_escapes_to_north_effect = yes
					usfp_underground_railroad_escapes_to_canada_effect = yes
					usfp_underground_railroad_escapes_to_canada_effect = yes  
				}
			}
		}
	}

	complete = {
		has_law = law_type:law_slavery_banned
		NOT = {
			has_journal_entry = je_acw_countdown
		}
		custom_tooltip = {
			text = csa_doesnt_exist
			OR = {
				NOT = { exists = c:CSA }
				c:CSA ?= {
					has_law = law_type:law_slavery_banned
				}
			}

		}
	}

	invalid = {
		OR = {
			country_has_voting_franchise = no
			has_law = law_type:law_council_republic
		}
	}
