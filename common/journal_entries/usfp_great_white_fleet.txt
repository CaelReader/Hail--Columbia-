je_usfp_the_new_navy = {
    icon = "gfx/interface/icons/invention_icons/destroyers.dds"

    group = je_group_historical_content

    is_shown_when_inactive = {
        c:USA ?= root
        OR = {
            has_technology_researched = ironclad_tech
            has_technology_researched = jeune_ecole
        }
    }

    possible = {
        #has modern ironclads researched
        has_technology_researched = ironclad_tech
        has_technology_researched = jeune_ecole
        any_country_in_pacific = {
            capital = {
                is_in_geographic_region = geographic_region_pacific
            }
            NOT = {
                this = root
                naval_power_projection >= {
                    value = root.naval_power_projection
                    multiply = 1.2
                }
            }
            #any ship is a Pre-Dreadnought
            any_scope_ship = {
                OR = {
                    type = ship_type:ship_type_pre_dreadnought
                    type = ship_type:ship_type_dreadnought
                    type = ship_type:ship_type_super_dreadnought
                }
            }
        }
    }

    immediate = {
        scope:journal_entry = {
            add_com_dlc_icon = {
                dlc = dlc_usfp
            }
        }
        ordered_country_in_pacific = {
            limit = {
                capital = {
                    is_in_geographic_region = geographic_region_pacific
                }
                NOT = {
                    this = root
                }
            }
            position = 0
            order_by = naval_power_projection
            save_scope_as = strongest_navy_in_pacific
        }
    }

    on_monthly_pulse = {
        effect = {
            ordered_country_in_pacific = {
                limit = {
                    capital = {
                        is_in_geographic_region = geographic_region_pacific
                    }
                    NOT = {
                        this = root
                    }
                }
                position = 0
                order_by = naval_power_projection
                save_scope_as = strongest_navy_in_pacific
            }
        }
    }

    complete = {
        #research pre-dreadnoughts
        has_technology_researched = destroyer
        #navy matches the strongest south american power
        trigger_if = {
            limit = {
                exists = scope:strongest_navy_in_pacific.naval_power_projection
            }
            naval_power_projection >= scope:strongest_navy_in_pacific.naval_power_projection
        }

        #has at least three pre-dreadnoughts
        any_scope_ship = {
            OR = {
                type = ship_type:ship_type_pre_dreadnought
                type = ship_type:ship_type_dreadnought
                type = ship_type:ship_type_super_dreadnought
            }
            count >= 3
        }
        OR = {
            has_law_or_variant = law_type:law_diplomatic_navy
            has_law_or_variant = law_type:law_professional_navy
        }
        #decommissioned wooden ships
        country_ship_type_fraction = {
            target = ship_type:ship_type_ship_of_the_line
            value < 0.10
        }
        country_ship_type_fraction = {
            target = ship_type:ship_type_early_ironclad
            value < 0.10
        }
        country_ship_type_fraction = {
            target = ship_type:ship_type_frigate
            value < 0.10
        }
        country_ship_type_fraction = {
            target = ship_type:ship_type_iron_frigate
            value < 0.10
        }
    }

    on_complete = {
        set_variable =  new_navy_completed
    }



    weight = 200
    should_be_pinned_by_default_uninvolved_or_context = yes
}
