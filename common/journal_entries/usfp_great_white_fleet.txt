je_usfp_the_new_navy = {
    icon = "gfx/interface/icons/invention_icons/monitor_tech.dds"

    group = je_group_historical_content

    is_shown_when_inactive = {
        c:USA ?= root
        has_technology_researched = jeune_ecole
    }

    possible = {
        #has modern ironclads researched
        OR = {
            #rival has a powerful navy
            any_rival_country = {
                naval_power_projection >= {
                    value = root.naval_power_projection
                    multiply = 1.2
                }
                #any ship is a Modern Ironclad or Pre-Dreadnought
            }
            #any other pacific power has a powerful navy
            any_country_in_americas = {
                NOT = {
                    this = root
                    naval_power_projection >= {
                        value = root.naval_power_projection
                        multiply = 1.2
                    }
                }
                #any ship is a Modern Ironclad or Pre-Dreadnought
            }
        }
    }

    immediate = {
        scope:journal_entry = {
            add_com_dlc_icon = {
                dlc = dlc_usfp
            }
        }
        ordered_country_in_americas = {
            limit = {
                NOT = {
                    this = root
                }
            }
            position = 0
            order_by = naval_power_projection
            save_scope_as = strongest_navy_in_americas
        }
    }

    on_monthly_pulse = {
        effect = {
            ordered_country_in_americas = {
                limit = {
                    NOT = {
                        this = root
                    }
                }
                position = 0
                order_by = naval_power_projection
                save_scope_as = strongest_navy_in_americas
            }
        }
    }

    complete = {
        #navy matches the strongest south american power
        naval_power_projection >= scope:strongest_navy_in_americas.naval_power_projection
        #has at least three pre-dreadnoughts
    }

    on_complete = {

    }

    on_fail = {

    }


    weight = 200
    should_be_pinned_by_default = yes
}
