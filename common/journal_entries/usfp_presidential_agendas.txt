usfp_je_agenda_jackson = {
	icon = "gfx/interface/icons/ideology_icons/ideology_leader/jackson_democrat.dds"

	group = je_group_historical_content

	immediate = {
		ruler = {
			save_scope_as = jackson_scope
		}
		scope:journal_entry = {
            add_com_dlc_icon = {
                dlc = dlc_usfp
            }
        }
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					tax_level > low
				 }
				 set_variable = {
					name = changed_taxes_above_low
					years = 10
				 }
			}
			if = {
				limit = { 
					OR = {
						NOT = {
							has_law_or_variant = law_type:law_legacy_slavery
						}
						is_enacting_law = law_type:law_slave_trade
						is_enacting_law = law_type:law_slavery_banned
						is_enacting_law = law_type:law_debt_slavery
						is_enacting_law = law_type:law_colonial_slavery
					}
				}
				set_variable = {
					name = attempted_to_change_slavery_law
					years = 10
				}
			}
		}
	}

	complete = {
		OR = {
			custom_tooltip = {
				text = all_presidential_agenda_items_complete
				in_election_campaign = yes
				has_law_or_variant = law_type:law_legacy_slavery
				has_law_or_variant = law_type:law_agrarianism
				has_law_or_variant = law_type:law_universal_suffrage
				NOT = {
					has_variable = changed_taxes_above_low
				}
				scaled_debt <= 0
				has_variable = usfp_indians_removed_end
			}
			ruler = {
				NOT = {
					has_ideology = ideology:ideology_jacksonian_democrat
				}
			}
		}
	}

	on_complete = {
		add_loyalists = {
			value = {
				value = 0.0
				if = {
					limit = {
						NOR = {
							NOT = {
								has_law_or_variant = law_type:law_legacy_slavery
							}
							is_enacting_law = law_type:law_slave_trade
							is_enacting_law = law_type:law_slavery_banned
							is_enacting_law = law_type:law_debt_slavery
							is_enacting_law = law_type:law_colonial_slavery
							has_variable = attempted_to_change_slavery_law
						}
					}
					add = 0.01
				}
				if = {
					limit = {
						has_law_or_variant = law_type:law_agrarianism
					}
					add = 0.01
				}
				if = {
					limit = {
						has_law_or_variant = law_type:law_universal_suffrage
					}
					add = 0.01
				}
				if = {
					limit = {
						NOT = {
							has_variable = changed_taxes_above_low
						}
					}
					add = 0.01
				}
				if = {
					limit = {
						scaled_debt <= 0
					}
					add = 0.01
				}
				if = {
					limit = {
						has_variable = usfp_indians_removed_end
					}
					add = 0.01
				}
			}
			interest_group = ig:ig_rural_folk
		}
		add_modifier = {
			name = usfp_agenda_completed_jackson_modifier
			years = 4
			multiplier = {
				value = {
					value = 0.0
					if = {
						limit = {
							NOR = {
								NOT = {
									has_law_or_variant = law_type:law_legacy_slavery
								}
								is_enacting_law = law_type:law_slave_trade
								is_enacting_law = law_type:law_slavery_banned
								is_enacting_law = law_type:law_debt_slavery
								is_enacting_law = law_type:law_colonial_slavery
								has_variable = attempted_to_change_slavery_law
							}
						}
						add = 1
					}
					if = {
						limit = {
							has_law_or_variant = law_type:law_agrarianism
						}
						add = 1
					}
					if = {
						limit = {
							has_law_or_variant = law_type:law_universal_suffrage
						}
						add = 1
					}
					if = {
						limit = {
							NOT = {
								has_variable = changed_taxes_above_low
							}
						}
						add = 1
					}
					if = {
						limit = {
							scaled_debt <= 0
						}
						add = 1
					}
					if = {
						limit = {
							has_variable = usfp_indians_removed_end
						}
						add = 1
					}
				}
			}
		}

	}

	invalid = {
		NOT = {
			has_law_or_variant = law_type:law_presidential_republic
		}
	}
	
	weight = 200
	should_be_pinned_by_default = yes
}

usfp_je_agenda_roosevelt_domestic = {
    icon = "gfx/interface/icons/invention_icons/investment_banks.dds"

    group = je_group_historical_content

    immediate = {
        ruler = {
            save_scope_as = roosevelt_scope
        }
        scope:journal_entry = {
            add_com_dlc_icon = {
                dlc = dlc_usfp
            }
        }
    }

    is_shown_when_inactive = {
        #activated by event on new ruler
    }

    on_monthly_pulse = {
        effect = {
            if = {
                limit = {
                    has_revolution = yes
                }
                set_variable = roosevelt_didnt_avoid_revolution
            }
        }
    }

    complete = {
        OR = {
            custom_tooltip = {
                text = all_presidential_agenda_items_complete
                in_election_campaign = yes
                #National Parks
                any_scope_building = {
                    is_building_type = building_usfp_national_park
                    count >= 10
                }
                #Busted Trusts
                has_variable = busted_trusts
                #Few Companies
                num_companies <= {
                    value = max_num_companies
                    subtract = 3
                }
                #Progressive Era Laws
                has_law_or_variant = law_type:law_interventionism
                institution_investment_level = {
                    institution = institution_workplace_safety
                    value >= 3
                }
                institution_investment_level = {
                    institution = institution_health_system
                    value >= 3
                }
                NOT = {
                    has_variable = roosevelt_didnt_avoid_revolution
                }
                #Marginalize Socialists
                NOT = {
                    any_political_movement = {
                        movement_is_socialist = yes
                        OR = {
                            political_movement_radicalism >= 0.75
                            political_movement_support >= 0.15
                        }
                    }
                }
                #Industrialists not powerful
                ig:ig_industrialists ?= {
                    is_powerful = no
                }
            }
            custom_tooltip = {
                text = roosevelt_is_not_president
                ruler ?= {
                    NOT = {
                        has_template = usa_roosevelt_character_template
                    }
                }
            }
        }
    }

    on_complete = {

    }

    invalid = {
        NOT = {
            has_law_or_variant = law_type:law_presidential_republic
        }
    }

    weight = 200
    should_be_pinned_by_default = yes
}

usfp_je_agenda_roosevelt_foreign = {
    icon = "gfx/interface/icons/invention_icons/breech_loading_artillery.dds"

    group = je_group_historical_content

    immediate = {
        ruler = {
            save_scope_as = roosevelt_scope
        }
        scope:journal_entry = {
            add_com_dlc_icon = {
                dlc = dlc_usfp
            }
        }
        set_variable = {
            name = roosevelt_starting_army
            value = army_power_projection
        }
    }

    is_shown_when_inactive = {
        #activated by event on new ruler
    }


    on_monthly_pulse = {

    }


    complete = {
        OR = {
            custom_tooltip = {
                text = all_presidential_agenda_items_complete
                in_election_campaign = yes
                #Cuba
                OR = {
                    c:CUB ?= {
                        is_subject_of = root
                    }
                    c:CUB ?= {
                        is_subject = no
                    }
                    AND = {
                        owns_entire_state_region = STATE_CENTRAL_CUBA
                        owns_entire_state_region = STATE_EASTERN_CUBA
                        owns_entire_state_region = STATE_WESTERN_CUBA
                    }
                }
                #Puerto Rico
                OR = {
                    c:PCO ?= {
                        is_subject_of = root
                    }
                    owns_entire_state_region = STATE_PUERTO_RICO
                }
                #Panama Canal
                any_scope_building = {
                    is_building_type = building_panama_canal
                    is_under_construction = no
                }
                #Monroe Doctrine enforced minus GBR
                any_country = {
                    filter = {
                        is_in_geographic_region = geographic_region_americas
                    }
                    percent = 0
                    is_subject = yes
                    overlord ?= {
                        NOR = {
                            is_in_geographic_region = geographic_region_americas
                            c:GBR ?= this
                        }
                    }
                }
                #Pacific Power Projection
                OR = {
                    any_subject_or_below = {
                        is_in_geographic_region = geographic_region_western_pacific
                        OR = {
                            is_subject_type = subject_type_colony
                            is_subject_type = subject_type_puppet
                        }
                    }
                    any_scope_state = {
                        is_in_geographic_region = geographic_region_western_pacific
                        count >= 3
                    }
                }
                #Army Reform
                AND = {
                    any_scope_building = {
                        filter = {
                            is_building_type = building_barrack
                        }
                        percent = 1
                        has_active_production_method = pm_advanced_tactics_training
                    }
                    army_power_projection >= usfp_roosevelt_army_target
                }
                #Expanded Power Bloc
                AND = {
                    is_power_bloc_leader = yes
                    num_power_bloc_members_script > 7
                    power_bloc.current_cohesion_percentage >= 0.90
                    power_bloc.power_bloc_rank >= 3
                }
                #Complete Great White Fleet
                has_variable = great_white_fleet_completed
            }
            custom_tooltip = {
                text = roosevelt_is_not_president
                ruler ?= {
                    NOT = {
                        has_template = usa_roosevelt_character_template
                    }
                }
            }
        }

    }

    on_complete = {
        custom_tooltip = extra_bonus_for_completing_all
    }

    invalid = {
        NOT = {
            has_law_or_variant = law_type:law_presidential_republic
        }
    }

    weight = 200
    should_be_pinned_by_default = yes
}
