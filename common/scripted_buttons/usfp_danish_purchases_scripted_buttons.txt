usfp_danish_purchase_send_offer_button = {
	name = "usfp_danish_purchase_send_offer_button"
	desc = "usfp_danish_purchase_send_offer_button_desc"

	visible = {
		always = yes
	}

	cooldown = {
		years = 12
	}


	possible = {
		scope:greenland_owner_scope ?= {
			relations:root >= relations_threshold:cordial
		}
		has_interest_marker_in_region = region_north_sea_coast
		is_at_war = no
		custom_tooltip = {
			text = danish_purchase_at_least_one
			OR = {
				has_variable = usfp_danish_purchases_greenland_desire
				has_variable = usfp_danish_purchases_iceland_desire
				has_variable = usfp_danish_purchases_west_indies_desire
			}
		}
	}

	effect = {
		custom_tooltip = send_offer_to_denmark
		show_as_tooltip = {
			usfp_danish_purchase_effect = { MULT = 1 }
		}
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				any_interest_group_in_government = {
					leader = {
						usfp_manifest_destiny_pro_expansion_trigger = yes
					}
				}
			}
			add = 100
		}
	}
}

				}
			}
			add = 100
		}
	}
}
