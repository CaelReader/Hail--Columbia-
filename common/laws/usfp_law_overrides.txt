### override just to hide these laws for the USA
INJECT:law_protectionism = {
	is_visible = {
		NOT = {
			has_journal_entry = je_usfp_american_constitution
		}
	}
}

INJECT:law_freedom_of_conscience = {
	is_visible = {
		NOT = {
			has_journal_entry = je_usfp_american_constitution
		}
	}
}

INJECT:law_ethnostate = {
    on_activate = {
        if = {
            limit = {
                c:USA ?= root
                country_has_primary_culture = cu:afro_american
            }
            remove_primary_culture = cu:afro_american
        }
        if = {
            limit = {
                c:USA ?= root
                NOT = {
                    country_has_primary_culture = cu:dixie
                }
            }
            add_primary_culture = cu:dixie
        }
    }
}
