usfp_long_reconstruction_progress_bar = {
	name = "usfp_long_reconstruction_progress_bar"
	desc = "usfp_long_reconstruction_progress_bar_desc"

	default_green = yes

	yearly_progress = {
		add = {
			desc = "afro_american_average_acceptance"
			value = usfp_long_reconstruction_get_average_afro_american_acceptance 
		}
		add = {
			desc = "afro_american_character_popularity"
			value = usfp_long_reconstruction_get_afro_american_popularity
		}
		add = {
			desc = "afro_american_movement_support"
			value = usfp_long_reconstruction_get_afro_american_movement_support
		}
		subtract = {
			desc = "dixie_radicalism"
			value = usfp_long_reconstruction_get_dixie_radicals
		}
		subtract = {
			desc = "supremacist_movement_support"
			value = usfp_long_reconstruction_get_supremacist_movement_support 
		}
		if = {
			limit = {
				any_political_movement = {
					is_political_movement_type = movement_usfp_neoconfederate
				}
			}
			subtract = {
				desc = "neoconfederate_movement_support"
				value = usfp_long_reconstruction_get_neoconfederate_movement_support
			}
		}
		if = {
			limit = {
				any_political_movement = {
					is_political_movement_type = movement_fascist
				}
			}
			subtract = {
				desc = "fascist_movement_support"
				value = usfp_long_reconstruction_get_fascist_movement_support
			}
		}

		if = {
			limit = {
				NOT = {has_global_variable = usfp_peculiar_reconciliation}
				has_global_variable = usfp_reconstruction_outcome_mixed
			}
			subtract = {
				desc = "usfp_reconstruction_outcome_mixed"
				add = 15
			}
		}
		if = {
			limit = {
				NOT = {has_global_variable = usfp_peculiar_reconciliation}
				has_global_variable = usfp_reconstruction_outcome_failure
			}
			subtract = {
				desc = "usfp_reconstruction_outcome_failure"
				add = 30
			}
		}
		if = {
			limit = {
				has_global_variable = usfp_peculiar_reconciliation
			}
			subtract = {
				desc = "usfp_peculiar_reconciliation"
				add = 20
			}
		}
		if = {
			limit = {
				any_scope_state = {
					has_modifier = usfp_jim_crow
				}
			}
			subtract = {
				desc = "afro_americans_under_jim_crow"
				add = usfp_long_reconstruction_get_afro_american_percent_under_jim_crow
			}
		}
	}
	start_value = 0
	min_value = 0
	max_value = 2200
 }

 usfp_reconstruction_northern_will_progress_bar = {
	name = "usfp_reconstruction_northern_will_progress_bar"
	desc = "usfp_reconstruction_northern_will_progress_bar_desc"
	default = yes

	start_value = 100
	min_value = 0
	max_value = 100

	monthly_progress = {
		subtract = {
			desc = "usfp_reconstruction_northern_will_base_decay"
			value = 0.5
		}
		add = {
			desc = "usfp_reconstruction_northern_will_radical_abolitionist_clout"
			value = {
				every_interest_group = {
					limit = {
						leader = {
							OR = {
								has_ideology = ideology:ideology_usfp_abolitionist_radical
								has_ideology = ideology:ideology_humanitarian
							}
						}
					}
					add = ig_clout
				}
			}
			
		}
		subtract = {
			desc = "supremacist_movement_support"
			value = {
				every_political_movement = {
					limit = {
						is_political_movement_type = movement_cultural_majority
					}
					add = this.political_movement_support
				}
			}
		}
		if = {
			limit = {
				taking_loans = yes
			}
			subtract = {
				desc = "size_of_loans"
				value = scaled_debt
			}
		}
		subtract = {
			desc = "[concept_turmoil]"
			value = country_turmoil
		}
		if = {
			limit = {
				authority < 0
			}
			add = {
				desc = "authority_deficit"
				value = relative_authority
			}
		}
		if = {
			limit = {
				any_scope_state = {
					has_decree = decree_usfp_radical_land_distribution
				}
			}
			subtract = {
				desc = "decree_usfp_radical_land_distribution"
				every_scope_state = {
					limit = {
						has_decree = decree_usfp_radical_land_distribution
					}
					add = 0.2
				}
			}
		}
	}
 }
