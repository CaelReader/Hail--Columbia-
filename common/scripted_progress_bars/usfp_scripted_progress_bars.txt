usfp_long_reconstruction_progress_bar = {
	name = "usfp_long_reconstruction_progress_bar"
	desc = "usfp_long_reconstruction_progress_bar_desc"

	default = yes

	yearly_progress = {
		add = {
			desc = "afro_american_average_acceptance"
			value = usfp_long_reconstruction_get_average_afro_american_acceptance 
		}
		add = {
			desc = "afro_american_character_popularity"
			value = usfp_long_reconstruction_get_afro_american_popularity
		}
		add = {
			desc = "afro_american_movement_support"
			value = usfp_long_reconstruction_get_afro_american_movement_support
		}
		subtract = {
			desc = "dixie_radicalism"
			value = usfp_long_reconstruction_get_dixie_radicals
		}
		subtract = {
			desc = "supremacist_movement_support"
			value = usfp_long_reconstruction_get_supremacist_movement_support 
		}
		if = {
			limit = {
				NOT = {has_global_variable = usfp_peculiar_reconciliation}
				has_global_variable = usfp_reconstruction_outcome_mixed
			}
			subtract = {
				desc = "usfp_reconstruction_outcome_mixed"
				add = 15
			}
		}
		if = {
			limit = {
				NOT = {has_global_variable = usfp_peculiar_reconciliation}
				has_global_variable = usfp_reconstruction_outcome_failure
			}
			subtract = {
				desc = "usfp_reconstruction_outcome_failure"
				add = 30
			}
		}
		if = {
			limit = {
				has_global_variable = usfp_peculiar_reconciliation
			}
			subtract = {
				desc = "usfp_peculiar_reconciliation"
				add = 20
			}
		}
		if = {
			limit = {
				any_scope_state = {
					has_modifier = usfp_jim_crow
				}
			}
			subtract = {
				desc = "afro_americans_under_jim_crow"
				add = usfp_long_reconstruction_get_afro_american_percent_under_jim_crow
			}
		}
	}
	start_value = 0
	min_value = 0
	max_value = 2000
 }
