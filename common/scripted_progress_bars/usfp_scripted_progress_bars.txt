	start_value = 100
	min_value = 0
	max_value = 100

	monthly_progress = {
		subtract = {
			desc = "usfp_reconstruction_northern_will_base_decay"
			value = 0.5
		}

			desc = "usfp_reconstruction_northern_will_radical_abolitionist_clout"

				every_interest_group = {
					limit = {
						leader = {
							OR = {
								has_ideology = ideology:ideology_usfp_abolitionist_radical
								has_ideology = ideology:ideology_humanitarian
							}
						}
					}
					add = ig_clout
				}
			}
			
		}
		subtract = {
			desc = "supremacist_movement_support"
			value = {
				every_political_movement = {
					limit = {
						is_political_movement_type = movement_cultural_majority
					}
					add = this.political_movement_support
				}
			}
		}
		if = {
			limit = {
				taking_loans = yes
			}
			subtract = {
				desc = "size_of_loans"
				value = scaled_debt
			}
		}
		if = {
			limit = {
				has_modifier = declared_bankruptcy
			}
			subtract = {
				desc = "declared_bankruptcy"
				value = 2
			}
		}
		subtract = {
			desc = "concept_turmoil"
			value = country_turmoil
		}
		if = {
			limit = {
				authority < 0
			}
			add = {
				desc = "authority_deficit"
				value = {
					add = authority
					divide = 200
				}
			}
		}
		if = {
			limit = {
				any_scope_state = {
					has_decree = decree_usfp_radical_land_distribution
				}
			}
			subtract = {
				desc = "decree_usfp_radical_land_distribution"
				every_scope_state = {
					limit = {
						has_decree = decree_usfp_radical_land_distribution
					}
					add = 0.2
				}
			}
		}
		if = {
			limit = {
				any_interest_group = {
					has_party = yes
					is_in_government = yes
					party = {
						OR = {
							is_party_type = conservative_party
							AND = {
								is_party_type = agrarian_party
								root = {
									NOT = {
										any_active_party = {
											is_party_type = conservative_party
										}
									}

							}
						}
					}
				}
			}
			subtract = {
				desc = "democratic_party_in_government"
				value = 1
			}
		}
	}
 }

 usfp_panic_1837_recovery_progress_bar = {
	name = "usfp_panic_1837_recovery_progress_bar"
	desc = "usfp_panic_1837_recovery_progress_bar_desc"
	default_green = yes

	start_value = 0
	min_value = 0
	max_value = 200

	monthly_progress = {
		#Natural recovery over time
		add = {
			value = 1
			desc = "panic_recovery_natural"
		}
		#GDP growth
		if = {
			limit = {
				var:panic_1837_last_month_gdp < var:panic_1837_this_month_gdp
			}
			add = {
				value = var:panic_1837_gdp_target
				divide = var:panic_1837_this_month_gdp
				subtract = 1
				multiply = 100
				max = 3
				desc = "panic_recovery_gdp_growth"
			}
		} else = {
			subtract = {
				value = var:panic_1837_last_month_gdp
				divide = var:panic_1837_this_month_gdp
				subtract = 1
				multiply = 100
				max = 3
				desc = "panic_recovery_gdp_growth"
			}
		}
		#Number of Gold Fields/Gold Mines
		add = {
			value = {
				every_scope_building = {
					limit = {
						OR = {
							is_building_group = bg_gold_fields
							is_building_group = bg_gold_mining
						}
					}
					add = level
				}
				divide = 5
			}
			desc = "panic_recovery_gold_sources"
		}
		#Number of Railroads
		add = {
			value = {
				every_scope_building = {
					limit = {
						is_building_type = building_railway
					}
					add = level
				}
				divide = 10
			}
			desc = "panic_recovery_railroads"
		}
		#Interventionism Law
		if = {
			limit = {
				has_law_or_variant = law_type:law_interventionism
			}
			add = {
				value = 0.5
				desc = "law_interventionism"
			}
		}
		#Bankruptcy/Default
		if = {
			limit = {
				OR = {
					in_default = yes
					has_modifier = declared_bankruptcy
				}
			}
			subtract = {
				value = 5
				desc = "panic_recovery_bankruptcy"
			}
		}
	}
 }
