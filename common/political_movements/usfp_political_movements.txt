movement_usfp_progressive = {
	category = movement_category_ideological
	ideology = ideology_usfp_progressive_movement
	active_law_radicalism_multiplier = 0.75 #they care a about a lot of laws
	character_ideologies = {
		ideology_usfp_progressive
		ideology_usfp_social_gospel_leader
	}
	creation_trigger = {
		c:USA ?= this
		has_global_variable = had_progressive_era
	}
	creation_weight = {
		value = 0 #event-spawned
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}
	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_usfp_progressive
			has_ideology = ideology:ideology_usfp_social_gospel_leader
			has_ideology = ideology:ideology_social_democrat
			has_ideology = ideology:ideology_feminist
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				has_ideology = ideology:ideology_usfp_progressive
			}
			multiply = {
				value = 5
			}
		}
	}
	pop_support_factors = {
		movement_support_high_urbanization
		movement_support_middle_below_expected_sol
		movement_support_high_literacy
		movement_support_academics
		movement_support_bureaucrats
		movement_support_engineers
		movement_support_shopkeepers
		movement_support_political_machine
		movement_support_pollution
	}
	
	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
					OR = {
						is_being_suppressed = yes
						is_being_bolstered = yes
					}
				}
			}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}	
		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}

		if = {
			limit = {
				strata = middle
			}
			if = {
				limit = {
					is_pop_type = shopkeepers
				}
				add = {
					value = 9
					desc = "POP_SHOPKEEPERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = bureaucrats
				}
				add = {
					value = 20
					desc = "POP_BUREAUCRATS"
				}
			} 
			else_if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 20
					desc = "POP_ACADEMICS"
				}
			} 
			else_if = {
				limit = {
					is_pop_type = engineers
				}
				add = {
					value = 25
					desc = "POP_ENGINEERS"
				}
			}
			else = {
				add = {
					value = 6
					desc = "MIDDLE_NO_ICON"
				}
			}
		}

		if = {
			limit = {
				state = {
					has_modifier = usfp_captive_state_government
				}
			}
			add = {
				value = 10 
				desc = "STATE_POLITICAL_MACHINE"
			}
		}

		if = {
			limit = {
				state = {
					state_region = {
						pollution_amount > 75
					}
				}
			}
			add = {
				value = 5 
				desc = "STATE_POLLUTED"
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
				strata = middle
			}
			add = { # Increases likelihood of support if impoverished
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		subtract = { #Radicalism pushes people out of Progressive Movement into other spheres
			desc = "POP_RADICALS"
			value = pop_radical_fraction
			multiply = 20
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		
		weight = {
			value = 1000
		}	
	
		state_weight = {
			value = 1		
		}
	}
		
	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 100
		}		
	}
}

			}

				value = 0.1

						is_being_bolstered = yes

					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}	
		add = {
			desc = "URBAN_STATE" # Increases support in rural states, decreases in urban
			value = 0.5
			subtract = state.state_urbanization_rate
			multiply = 10
		}

		if = {
			limit = {
				culture = cu:dixie
			}
			add = {
				value = 15
				desc = "POP_SUPPORT_DIXIE_CULTURE"
			}
		}

		if = {
			limit = {
				is_pop_type = aristocrats
			}
			add = {
				value = 30
				desc = "POP_ARISTOCRATS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = farmers
			}
			add = {
				value = 10
				desc = "POP_FARMERS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = peasants
			}
			add = {
				value = 15
				desc = "POP_PEASANTS"
			}
		}
		else_if = {
			limit = {
				is_pop_type = shopkeepers
			}
			add = {
				value = 5
				desc = "POP_SHOPKEEPERS"
			}
		}
		if = {
			limit = {
				has_global_variable = usfp_reconstruction_outcome_radical
			}
			add = {
				desc = "POP_SUPPORT_RADICAL_RECONSTRUCTION"
				value = 10
			}
		} else_if = {
			limit = {
				has_global_variable = usfp_reconstruction_outcome_mixed
			}
			add = {
				desc = "POP_SUPPORT_MIXED_RECONSTRUCTION"
				value = 5
			}
		}

		if = {
			limit = {
				state = {
					has_modifier = usfp_jim_crow
				}
			}
			add = {
				desc = "POP_SUPPORT_JIM_CROW_STATE"
				value = 10
			}
		}

		if = {
			limit = {
				state = {
					any_scope_culture = {
						has_discrimination_trait_group = heritage_group_african
					}
				}
			}
			add = {
				desc = "POP_SUPPORT_BLACK_SOL_DIFFERENCE"
				value = {
					state = {
						ordered_scope_pop = {
							limit = {
								culture = {
									has_discrimination_trait_group = heritage_group_african
								}
							}
							order_by = standard_of_living
							position = 0
							value = standard_of_living
						}
					}
				}
				subtract = standard_of_living
				max = 50
			}
		}


		if = {
			limit = {
				standard_of_living < state.average_expected_sol
			}
			add = { # Increases likelihood of support if impoverished
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}

		add = { # Weakened by increasing literacy rates
			desc = "POP_LITERACY"
			this = {
				value = literacy_rate
				multiply = -2
			}
		}

		multiply = { # Multiplied by 2x max
			desc = "POP_RADICALS"
			value = pop_radical_fraction
			add = 1
		}

		if = { # If a more radical movement exists, it'll poach radicals from this one.
			limit = {
				owner ?= {
					any_political_movement = {
						is_political_movement_type = movement_fascist
					}
				}
			}
			multiply = {
				desc = "POP_RADICAL_MOVEMENT_LOSSES"
				value = 1
				subtract = pop_radical_fraction
			}
		}	
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		
		weight = {
			value = 1000
		}	
	
		state_weight = {
			value = 1		
		}
	}
		
	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 100
		}		
	}
}
