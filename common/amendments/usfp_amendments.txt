    parent = law_consumption_based_taxation # optional reference to a law that dictates what stances IGs and movements have toward this amendment

    allowed_laws = { # list of laws that the amendment can be added to
		law_elected_bureaucrats
		law_appointed_bureaucrats
    }

    modifier = {
		country_institution_size_change_speed_mult = -0.25
		country_usfp_spoil_system_on_reform_modifier = -0.25
	}

    possible = {
        has_journal_entry = je_usfp_american_constitution
    }
	would_sponsor = {
		always = no
	}
   	can_repeal = {
		legitimacy > 60
		OR = {
			trigger_if = {
				limit = {
					has_variable = usfp_spoils_system_abolish_var
				}
				custom_tooltip = {
					text = spoils_system_can_be_abolished_tt
					var:usfp_spoils_system_abolish_var >= 5
				}
			}
			NOT = {
				has_journal_entry = je_usfp_american_constitution
			}
			any_interest_group_in_government = {
				is_powerful = yes
				leader = {
					OR = {
						has_ideology = ideology:ideology_usfp_progressive
						has_ideology = ideology:ideology_social_democrat
					}
				}
			}
		}
	}

    ai_will_revoke = {
        always = yes
    }

	amendment_activism_multiplier = 0.0
}
