amendment_usfp_gag_rule = {
    parent = law_legacy_slavery

    allowed_laws = { # list of laws that the amendment can be added to
		law_legacy_slavery
    }

    modifier = {
		interest_group_ig_intelligentsia_approval_add = -3
		interest_group_ig_intelligentsia_pol_str_mult = -0.10
		interest_group_ig_landowners_approval_add = 4
		political_movement_radicalism_add = 0.05
	}

    possible = {
        always = yes
    }

	can_repeal = {
		OR = {
			any_revolution_ongoing = yes
			AND = {
				ig:ig_landowners = {
					is_in_government = no
				}
				government_legitimacy >= 51
			}
			AND = {
				ig:ig_intelligentsia = {
					is_marginal = yes
				}
				OR = {
					any_interest_group = {
						is_powerful = yes
						OR = {
							has_ideology = ideology:ideology_pro_slavery
							leader = {
								has_ideology = ideology:ideology_slaver
							}
						}
					}
					ruler = {
						has_ideology = ideology:ideology_slaver
					}
				}
				government_legitimacy >= 51
			}
		}

	}

    ai_will_revoke = {
        always = yes
    }
}
