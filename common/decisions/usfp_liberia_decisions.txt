usfp_liberia_declare_independence = {
	is_shown = {
		exists = c:LIB
		this = c:LIB
		is_subject = yes
		has_variable = american_colonization_society
		has_law_or_variant = law_type:law_presidential_republic
	}

	possible = {
		OR = {
			has_technology_researched = nationalism
			has_law_or_variant = law_type:law_landed_voting
			has_law_or_variant = law_type:law_wealth_voting
			has_law_or_variant = law_type:law_census_voting
			has_law_or_variant = law_type:law_universal_suffrage
		}
		is_diplomatic_play_committed_participant = no
	}

	when_taken = {
		overlord = {
			trigger_event = usfp_liberia.1
		}
		if = {
			limit = { 
				country_has_voting_franchise = no
			}
			activate_law = law_type:law_landed_voting
		}
		remove_variable = american_colonization_society
	}

	ai_chance = {
		value = 100
	}
}
