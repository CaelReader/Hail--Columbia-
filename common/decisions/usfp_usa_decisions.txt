dismantle_jim_crow = {
	is_shown = {
		any_scope_state = {
			has_modifier = usfp_jim_crow
		}
	}

	possible = {
		OR = {
			#New Africa usage
			country_has_primary_culture = cu:afro_american
			AND = {
				OR = {
					NOT = {
						has_law = law_type:law_presidential_republic
					}
					has_law = law_type:law_multicultural
				}
				OR = {
					institution_investment_level = {
						institution = institution_home_affairs
						value >= 2
					}
					institution_investment_level = {
						institution = institution_police
						value >= 2
					}
				}
				authority >= 300
				any_interest_group = {
					is_in_government = yes
					law_stance = {
						law = law_type:law_racial_segregation
						value < neutral
					}
				}
			}
		}
	}

	when_taken = {
		add_modifier = {
			name = usfp_enforcing_reconstruction
			multiplier = 4
			years = 5
		}
		every_scope_state = {
			limit = {
				has_modifier = usfp_jim_crow
			}
			remove_modifier = usfp_jim_crow
			add_radicals_in_state = {
				value = very_large_radicals
				culture = cu:dixie
			}
			add_loyalists_in_state = {
				value = large_radicals
				culture = cu:afro_american
			}
		}
		every_scope_state = {
			limit = {
				has_modifier = usfp_white_state_government_modifier
			}
			remove_modifier = usfp_white_state_government_modifier
		}
	}

	ai_chance = {
		value = 1000
	}
}

negotiate_mexican_border = {
	is_shown = {
		ROOT = c:USA
		OR = {
			has_state_in_state_region = STATE_COLORADO
			has_state_in_state_region = STATE_NEW_MEXICO
		}
		NOR = {
			has_state_in_state_region = STATE_ARIZONA
			has_state_in_state_region = STATE_CALIFORNIA
			has_state_in_state_region = STATE_NEVADA
		}
		exists = c:MEX
		c:MEX = {
			OR = {
				has_state_in_state_region = STATE_COLORADO
				has_state_in_state_region = STATE_NEW_MEXICO
			}
			is_player = no
		}
		s:STATE_COLORADO = {
			NOT = {
				any_scope_state = {
					NOR = {
						owner = c:MEX
						owner = c:USA
					}
				}
			}
		}
		s:STATE_NEW_MEXICO = {
			NOT = {
				any_scope_state = {
					NOR = {
						owner = c:MEX
						owner = c:USA
					}
				}
			}
		}
		NOT = {
			has_variable = mexican_border_negotiated
		}
	}

	possible = {
		OR = {
			c:MEX = {
				is_subject_of = c:USA
			}
			c:MEX = {
				relations:root >= relations_threshold:friendly
			}
		}
		is_diplomatic_play_committed_participant = no
		c:MEX = {
			is_diplomatic_play_committed_participant = no
		}
	}

	when_taken = {
		add_treasury = -1000000
		c:MEX = {
			add_treasury = 1000000
		}
		if = {
			limit = { 
				c:MEX = {
					has_state_in_state_region = STATE_COLORADO
				}
			}
			s:STATE_COLORADO.region_state:MEX = {
				set_state_owner = c:USA
			}
		}
		if = {
			limit = { 
				c:MEX = {
					has_state_in_state_region = STATE_NEW_MEXICO
				}
			}
			hidden_effect = {
				s:STATE_NEW_MEXICO.region_state:MEX = {
					set_state_owner = c:USA
				}
			}
			s:STATE_NEW_MEXICO = {
				set_owner_of_provinces = {
					country = c:MEX
					provinces = { xC03080 xB566BA x17660E x9C15D3 x53F9F0 xFB0FA8 x2DF849 xBD8033 x17E6C3 x808AB5 x168663 xA84AFD x0DAC3E x3CC82C
						x67B298 xA156C7 x851687 x90EA7D x650434 x5DA018 x46C455 xB6EC9A x940839 x84CE0F x4025EE x37B6CF x9C0FED xEF029F xAA2075 }
				}
			}
		}
		if = {
			limit = { 
				c:MEX = {
					has_state_in_state_region = STATE_TEXAS
				}
				c:USA = {
					has_state_in_state_region = STATE_TEXAS
				}
				NOT = {
					exists = c:TEX
				}
			}
			s:STATE_TEXAS.region_state:MEX = {
				set_state_owner = c:USA
			}
			add_treasury = -1000000
			c:MEX = {
				add_treasury = 1000000
			}
		}
		set_variable = mexican_border_negotiated
	}

	ai_chance = {
		value = 100
	}
}

usfp_promote_afro_american_migration_africa = {
	is_shown = {
		c:USA ?= root
		usfp_is_afro_american_discriminated = yes
		institution_investment_level = {
			institution = institution_colonial_affairs
			value >= 1
		}
	}

	possible = {
		usfp_is_afro_american_discriminated = yes
		OR = {
			c:LIB ?= {
				is_subject_of = root
			}
			any_scope_state = {
				state_is_in_africa = yes
				NOT = {
					state_region = s:STATE_SINAI
				}
			}
		}
		any_scope_state = {
			state_is_in_north_america = yes
			culture_percent_state = {
				target = cu:afro_american
				value > 0.15
			}
		}
		institution_investment_level = {
			institution = institution_colonial_affairs
			value >= 2
		}
		is_at_war = no
		custom_tooltip = {
			text = usfp_afro_american_colonization_promoted
			NOT = {
				has_variable = usfp_afro_american_colonization_promoted
			}
		}
	}

	when_taken = {
		add_treasury = -100000
		set_variable = {
			name = usfp_afro_american_colonization_promoted
			years = 5
		}
		if = {
			limit = { 
				any_scope_state = {
					state_is_in_africa = yes
					NOT = {
						state_region = s:STATE_SINAI
					}
				}
			}
			ordered_scope_state = {
				limit = {
					state_is_in_africa = yes
					NOT = {
						state_region = s:STATE_SINAI
					}
				}
				order_by = free_arable_land
				save_scope_as = afro_american_colonization_target_state
			}
		}
		else = {
			c:LIB ?= {
				capital = {
					save_scope_as = afro_american_colonization_target_state
				}
			}
		}
		usfp_afro_american_colonization_effect = yes
		add_radicals = {
			value = small_radicals
			culture = cu:afro_american
		}
		add_loyalists = {
			value = small_radicals
			interest_group = ig:ig_petty_bourgeoisie
		}
	}
	
	ai_chance = {
		value = 0
		if = {
			limit = {
				c:LIB ?= {
					is_player = yes
				}
			}
			add = 10
		}
	}
}
