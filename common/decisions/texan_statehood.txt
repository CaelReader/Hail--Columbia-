texan_statehood_tex = {
	is_shown = {
		exists = c:TEX
		root = c:TEX
		root = {
			is_subject = no
			is_at_war = no
		}
	}

	possible = {
		NOT = {
			has_variable = texas_asked_statehood
		}
		c:USA = {
			relations:root >= relations_threshold:cordial
			is_at_war = no
		}
	}

	when_taken = {
		trigger_event = texan_war_of_independence.1
		set_variable = {
			name = texas_asked_statehood
			years = 4
		}
	}

	ai_chance = {
		base = 100
	}
}
